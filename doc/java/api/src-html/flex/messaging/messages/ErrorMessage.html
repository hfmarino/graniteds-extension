021    package flex.messaging.messages;
022    
023    import org.granite.messaging.service.ServiceException;
024    import org.granite.messaging.service.security.SecurityServiceException;
025    
026    import java.io.PrintWriter;
027    import java.io.StringWriter;
028    import java.util.HashMap;
029    import java.util.Map;
030    
031    /**
032     * @author Franck WOLFF
033     */
034    public class ErrorMessage extends AcknowledgeMessage {
035    
036        private static final long serialVersionUID = 1L;
037    
038        private String faultCode = "Server.Call.Failed";
039        private String faultDetail;
040        private String faultString;
041        private Object rootCause;
042        private Map<String, Object> extendedData;
043    
044        private transient boolean loginError = false;
045    
046        public ErrorMessage() {
047            super();
048        }
049    
050        public ErrorMessage(Throwable t) {
051            super();
052            init(t);
053        }
054    
055        public ErrorMessage(Message request) {
056            this(request, false);
057        }
058    
059        public ErrorMessage(Message request, Throwable t) {
060            this(request, t, false);
061        }
062    
063        public ErrorMessage(Message request, boolean keepClientId) {
064            super(request, keepClientId);
065        }
066    
067        public ErrorMessage(Message request, Throwable t, boolean keepClientId) {
068            super(request, keepClientId);
069            if (request instanceof CommandMessage) {
070                loginError = (
071                    ((CommandMessage)request).isLoginOperation() &&
072                    (t instanceof SecurityServiceException)
073                );
074            }
075            init(t);
076        }
077    
078        private void init(Throwable t) {
079            if (t instanceof ServiceException) {
080                ServiceException se = (ServiceException)t;
081    
082                this.faultCode = se.getCode();
083                this.faultString = se.getMessage();
084    
085                if (t instanceof SecurityServiceException)
086                    this.faultDetail = se.getDetail();
087                else
088                    this.faultDetail = se.getDetail() + getStackTrace(t);
089                
090                this.extendedData = se.getExtendedData();
091            }
092            else if (t != null) {
093                this.faultString = t.getMessage();
094                this.faultDetail = getStackTrace(t);
095            }
096    
097            if (!(t instanceof SecurityServiceException)) {
098                for (Throwable root = t; root != null; root = root.getCause())
099                    rootCause = root;
100            }
101        }
102    
103        private String getStackTrace(Throwable t) {
104            StringWriter sw = new StringWriter();
105            t.printStackTrace(new PrintWriter(sw));
106            return sw.toString().replace("\r\n", "\n").replace('\r', '\n');
107        }
108    
109        public String getFaultCode() {
110            return faultCode;
111        }
112        public void setFaultCode(String faultCode) {
113            this.faultCode = faultCode;
114        }
115    
116        public String getFaultDetail() {
117            return faultDetail;
118        }
119        public void setFaultDetail(String faultDetail) {
120            this.faultDetail = faultDetail;
121        }
122    
123        public String getFaultString() {
124            return faultString;
125        }
126        public void setFaultString(String faultString) {
127            this.faultString = faultString;
128        }
129    
130        public Map<String, Object> getExtendedData() {
131            return extendedData;
132        }
133        public void setExtendedData(Map<String, Object> extendedData) {
134            this.extendedData = extendedData;
135        }
136    
137        public Object getRootCause() {
138            return rootCause;
139        }
140        public void setRootCause(Object rootCause) {
141            this.rootCause = rootCause;
142        }
143    
144        public boolean loginError() {
145            return loginError;
146        }
147    
148        public ErrorMessage copy(Message request) {
149            ErrorMessage copy = new ErrorMessage(request, null);
150            copy.faultCode = faultCode;
151            copy.faultDetail = faultDetail;
152            copy.faultString = faultString;
153            copy.loginError = loginError;
154            copy.rootCause = rootCause;
155            copy.extendedData = new HashMap<String, Object>(extendedData);
156            return copy;
157        }
158    
159        @Override
160        public String toString() {
161            return toString("");
162        }
163    
164        @Override
165        public String toString(String indent) {
166            StringBuilder sb = new StringBuilder(512);
167            sb.append(getClass().getName()).append(" {");
168            sb.append('\n').append(indent).append("  faultCode = ").append(faultCode);
169            sb.append('\n').append(indent).append("  faultDetail = ").append(faultDetail);
170            sb.append('\n').append(indent).append("  faultString = ").append(faultString);
171            sb.append('\n').append(indent).append("  rootCause = ").append(rootCause);
172            sb.append('\n').append(indent).append("  extendedData = ").append(extendedData);
173            super.toString(sb, indent, null);
174            sb.append('\n').append(indent).append('}');
175            return sb.toString();
176        }
177    }
