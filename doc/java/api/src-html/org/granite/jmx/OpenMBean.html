064    /**
065     * The OpenMBean class wraps an instance of any bean and introspects its properties and methods
066     * for MBeanAttributes and MBeanOperations. It implements all functionalities required by a
067     * {@link DynamicMBean} and returns an OpenMBeanInfo object.
068     * <br>
069     * <br>
070     * Limitations:
071     * <ul>
072     * <li>only attributes and operations are supported (no contructor and no notification),</li>
073     * <li>only {@link SimpleType} and {@link ArrayType} are supported (no composite type).</li>
074     * </ul>
075     * 
076     * @author Franck WOLFF
077     */
078    @SuppressWarnings({ "unchecked", "rawtypes" })
079    public class OpenMBean implements DynamicMBean {
080

120            /**
121             * Creates a new OpenMBean instance and instrospects its child class for attributes
122             * and operations.
123             */
124            protected OpenMBean() {
125                    this.info = init(getClass(), OpenMBean.class, attributesMap, operationsMap);
126                    this.instance = this;
127            }
128            
129            private OpenMBean(Class<?> beanClass, Class<?> stopClass, Object instance) {
130                    this.info = init(beanClass, stopClass, attributesMap, operationsMap);
131                    this.instance = instance;
132            }

134            ///////////////////////////////////////////////////////////////////////////
135            // Static OpenMBean creators.
136    
137            /**
138             * Creates a new OpenMBean by introspecting and wrapping the <tt>instance</tt> parameter.
139             * This method search for an interface named <tt>instance.getClass().getSimpleName() + "MBean"</tt>
140             * and, if it finds it, uses it for introspection. Otherwise, the class of the <tt>instance</tt>
141             * object is used instead.
142             * 
143             * @param instance an instance of a bean to introspect.
144             * @return a new OpenMBean instance that wraps the instance bean.
145             */
146            public static OpenMBean createMBean(Object instance) {
147                    
148                    Class<?> beanClass = instance.getClass();
149                    String interMBeanName = beanClass.getSimpleName() + "MBean";
150                    for (Class<?> inter : instance.getClass().getInterfaces()) {
151                            if (interMBeanName.equals(inter.getSimpleName())) {
152                                    beanClass = inter;
153                                    break;
154                            }
155                    }
156                    
157                    Class<?> stopClass = null;
158                    if (!beanClass.isInterface()) {
159                            stopClass = beanClass.getSuperclass();
160                            if (stopClass == null)
161                                    stopClass = Object.class;
162                    }
163                    
164                    return new OpenMBean(beanClass, stopClass, instance);
165            }
166            
167            /**
168             * Creates a new OpenMBean by introspecting the <tt>beanClass</tt> parameter and wrapping
169             * the <tt>instance</tt> parameter.
170             * 
171             * @param beanClass a class (or interface) used for introspection.
172             * @param instance the bean to encapsulate.
173             * @return a new OpenMBean instance that wraps the instance bean.
174             * @throws IllegalArgumentException if instance is not an instance of beanClass.
175             */
176            public static OpenMBean createMBean(Class<?> beanClass, Object instance) {
177                    if (!beanClass.isAssignableFrom(instance.getClass()))
178                            throw new IllegalArgumentException("Instance " + instance + " should be an instance of " + beanClass);
179    
180                    Class<?> stopClass = null;
181                    if (!beanClass.isInterface()) {
182                            stopClass = beanClass.getSuperclass();
183                            if (stopClass == null)
184                                    stopClass = Object.class;
185                    }
186    
187                    return new OpenMBean(beanClass, stopClass, instance);
188            }

190            ///////////////////////////////////////////////////////////////////////////
191            // Static initialization methods.
192            
193            private static MBeanInfo init(
194                    Class<?> beanClass,
195                    Class<?> stopClass,
196                    Map<String, PropertyDescriptor> attributesMap,
197                    Map<String, MethodDescriptor> operationsMap) {
198                    
199                    MBean mb = beanClass.getAnnotation(MBean.class);
200                    
201                    String description = null;
202                    if (mb != null)
203                            description = mb.description();
204                    if (description == null)
205                            description = beanClass.getSimpleName() + " MBean";
206                    
207                    List<OpenMBeanAttributeInfo> attributes = new ArrayList<OpenMBeanAttributeInfo>();
208                    List<OpenMBeanOperationInfo> operations = new ArrayList<OpenMBeanOperationInfo>();
209                    
210                    try {
211                            BeanInfo beanInfo = (
212                                    stopClass == null ?
213                                    Introspector.getBeanInfo(beanClass) :
214                                    Introspector.getBeanInfo(beanClass, stopClass)
215                            );
216                            
217                            Set<Method> attributeMethods = new HashSet<Method>();
218                            
219                            // Attributes.
220                            for (PropertyDescriptor property : beanInfo.getPropertyDescriptors()) {
221    
222                                    MBeanAttribute mba = null;
223                                    if (property.getReadMethod() != null)
224                                            mba = property.getReadMethod().getAnnotation(MBeanAttribute.class);
225                                    if (mba == null && property.getWriteMethod() != null)
226                                            mba = property.getWriteMethod().getAnnotation(MBeanAttribute.class);
227                                    
228                                    String name = property.getName();
229                                    
230                                    String desc = null;
231                                    if (mba != null)
232                                            desc = mba.description();
233                                    if (desc == null)
234                                            desc = name.substring(0, 1).toUpperCase() + name.substring(1) + " Attribute";
235                                    
236                                    OpenType type = getOpenType(property.getPropertyType());
237                                    
238                                    attributes.add(new OpenMBeanAttributeInfoSupport(
239                                            name,
240                                            desc,
241                                            type,
242                                            property.getReadMethod() != null,
243                                            property.getWriteMethod() != null,
244                                            property.getReadMethod() != null && property.getReadMethod().getName().startsWith("is")
245                                    ));
246                                    attributesMap.put(property.getName(), property);
247                                    
248                                    if (property.getReadMethod() != null)
249                                            attributeMethods.add(property.getReadMethod());
250                                    if (property.getWriteMethod() != null)
251                                            attributeMethods.add(property.getWriteMethod());
252                            }
253                            
254                            // Operations
255                            for (MethodDescriptor method : beanInfo.getMethodDescriptors()) {
256                                    
257                                    if (attributeMethods.contains(method.getMethod()))
258                                            continue;
259                                    
260                                    MBeanOperation mbo = method.getMethod().getAnnotation(MBeanOperation.class);
261                                    
262                                    String name = method.getName();
263    
264                                    String desc = null;
265                                    if (mbo != null)
266                                            desc = mbo.description();
267                                    if (desc == null)
268                                            desc = name.substring(0, 1).toUpperCase() + name.substring(1) + " Operation";
269                                    
270                                    List<OpenMBeanParameterInfo> parameters = new ArrayList<OpenMBeanParameterInfo>();
271                                    Annotation[][] annotations = method.getMethod().getParameterAnnotations();
272                                    
273                                    int i = 0;
274                                    for (Class<?> parameter : method.getMethod().getParameterTypes()) {
275                                            String paramName = getParameterName(annotations, i);
276                                            String paramDesc = getParameterDescription(annotations, i);
277                                            OpenType paramType = getOpenType(parameter);
278                                            
279                                            parameters.add(new OpenMBeanParameterInfoSupport(paramName, paramDesc, paramType));
280                                            
281                                            i++;
282                                    }
283                                    
284                                    OpenType returnedType = getOpenType(method.getMethod().getReturnType());
285                                    
286                                    int impact = MBeanOperationInfo.UNKNOWN;
287                                    if (mbo != null && mbo.impact() != null) {
288                                            switch (mbo.impact()) {
289                                                    case ACTION: impact = MBeanOperationInfo.ACTION; break;
290                                                    case ACTION_INFO: impact = MBeanOperationInfo.ACTION_INFO; break;
291                                                    case INFO: impact = MBeanOperationInfo.INFO; break;
292                                            }
293                                    }
294                                    
295                                    operations.add(new OpenMBeanOperationInfoSupport(
296                                            name,
297                                            desc,
298                                            parameters.toArray(new OpenMBeanParameterInfo[parameters.size()]),
299                                            returnedType,
300                                            impact
301                                    ));
302                                    
303                                    String[] paramClasses = new String[parameters.size()];
304                                    for (i = 0; i < parameters.size(); i++)
305                                            paramClasses[i] = parameters.get(i).getOpenType().getTypeName().toString();
306                                    String qName = name + Arrays.toString(paramClasses);
307                                    
308                                    operationsMap.put(qName, method);
309                            }
310                    }
311                    catch (Exception e) {
312                            throw new RuntimeException("Could not introspect MBean class: " + beanClass, e);
313                    }
314                    
315                    return new OpenMBeanInfoSupport(
316                            beanClass.getName(),
317                            description,
318                            attributes.toArray(new OpenMBeanAttributeInfo[attributes.size()]),
319                            new OpenMBeanConstructorInfo[0],
320                            operations.toArray(new OpenMBeanOperationInfo[operations.size()]),
321                            new MBeanNotificationInfo[0]
322                    );
323            }

325            private static String getParameterName(Annotation[][] paramAnnotations, int index) {
326                    String name = null;
327                    if (paramAnnotations != null && paramAnnotations.length > index) {
328                            for (Annotation annot : paramAnnotations[index]) {
329                                    if (MBeanParameter.class.equals(annot.annotationType())) {
330                                            name = ((MBeanParameter)annot).name();
331                                            break;
332                                    }
333                            }
334                    }
335                    return (name != null ? name : "arg" + (index + 1));
336            }
337            
338            private static String getParameterDescription(Annotation[][] paramAnnotations, int index) {
339                    String description = null;
340                    if (paramAnnotations != null && paramAnnotations.length > index) {
341                            for (Annotation annot : paramAnnotations[index]) {
342                                    if (MBeanParameter.class.equals(annot.annotationType())) {
343                                            description = ((MBeanParameter)annot).description();
344                                            break;
345                                    }
346                            }
347                    }
348                    return (description != null ? description : "Operation Parameter " + (index + 1));
349            }
350            
351            private static OpenType getOpenType(Class<?> clazz) throws OpenDataException {
352                    
353                    if (SIMPLE_TYPES.containsKey(clazz))
354                            return SIMPLE_TYPES.get(clazz);
355                    
356                    if (clazz.isArray()) {
357                            int dimension = 1;
358                            Class<?> componentType = clazz.getComponentType();
359                            while (componentType.isArray()) {
360                                    dimension++;
361                                    componentType = componentType.getComponentType();
362                            }
363                            return new ArrayType(dimension, getOpenType(componentType));
364                    }
365                    
366                    throw new OpenDataException("Unsupported type: " + clazz);
367            }

372            /**
373             * {@inheritDoc}
374             */
375            public synchronized Object getAttribute(String attribute)
376                    throws AttributeNotFoundException, MBeanException, ReflectionException {
377                    
378                    PropertyDescriptor property = attributesMap.get(attribute);
379                    if (property != null && property.getReadMethod() != null) {
380                            try {
381                                    return property.getReadMethod().invoke(instance);
382                            }
383                            catch (Exception e) {
384                                    throw new ReflectionException(e, "Could not get attribute value: " + attribute);
385                            }
386                    }
387                    throw new AttributeNotFoundException("Attribute " + attribute + " not found");
388            }
389    
390            /**
391             * {@inheritDoc}
392             */
393            public synchronized AttributeList getAttributes(String[] names) {
394                    AttributeList attributes = new AttributeList();
395                    for (String name : names) {
396                            try {
397                                    attributes.add(new Attribute(name, getAttribute(name)));
398                            }
399                            catch (Exception e) {
400                                    // ignore...
401                            }
402                    }
403                    return attributes;
404            }
405    
406            /**
407             * {@inheritDoc}
408             */
409            public synchronized MBeanInfo getMBeanInfo() {
410                    return info;
411            }
412    
413            /**
414             * {@inheritDoc}
415             */
416            public synchronized Object invoke(String actionName, Object[] params, String[] signature)
417                    throws MBeanException, ReflectionException {
418                    
419                    if (signature == null)
420                            signature = new String[0];
421                    
422                    String qName = actionName + Arrays.toString(signature);
423                    
424                    MethodDescriptor method = operationsMap.get(qName);
425                    if (method == null)
426                            throw new RuntimeException("Method not found: " + qName);
427    
428                    try {
429                            return method.getMethod().invoke(instance, params);
430                    }
431                    catch (Exception e) {
432                            throw new ReflectionException(e, "Could not invoke operation: " + qName);
433                    }
434            }
435    
436            /**
437             * {@inheritDoc}
438             */
439            public synchronized void setAttribute(Attribute attribute)
440                    throws AttributeNotFoundException, InvalidAttributeValueException, MBeanException, ReflectionException {
441                    
442                    PropertyDescriptor property = attributesMap.get(attribute.getName());
443                    if (property != null && property.getWriteMethod() != null) {
444                            try {
445                                    property.getWriteMethod().invoke(instance, attribute.getValue());
446                            }
447                            catch (Exception e) {
448                                    throw new ReflectionException(e, "Could not set attribute value: " + attribute.getName() + "=" + attribute.getValue());
449                            }
450                    }
451                    else
452                            throw new AttributeNotFoundException("Attribute " + attribute.getName() + " not found");
453            }
454    
455            /**
456             * {@inheritDoc}
457             */
458            public synchronized AttributeList setAttributes(AttributeList attributes) {
459                    AttributeList returnedAttributes = new AttributeList();
460                    for (Object a : attributes) {
461                            Attribute attribute = (Attribute)a;
462                            try {
463                                    setAttribute(attribute);
464                                    returnedAttributes.add(new Attribute(attribute.getName(), getAttribute(attribute.getName())));
465                            }
466                            catch (Exception e) {
467                                    // Ignore...
468                            }
469                    }
470                    return returnedAttributes;
471            }
