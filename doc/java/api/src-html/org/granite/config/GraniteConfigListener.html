060    public class GraniteConfigListener implements ServletContextListener {

062        private static final String GRANITE_CONFIG_SHUTDOWN_KEY = GraniteConfig.class.getName() + "_SHUTDOWN";
063        public static final String GRANITE_CONFIG_ATTRIBUTE = "org.granite.config.flexFilter";
064        public static final String GRANITE_MBEANS_ATTRIBUTE = "registerGraniteMBeans";
065    
066        private static final Logger log = Logger.getLogger(GraniteConfigListener.class);
067    
068        public void contextInitialized(ServletContextEvent sce) {
069            try {
070                ServletContext context = sce.getServletContext();
071    
072                log.info("Initializing GraniteDS...");
073                
074                Class<?> flexFilterClass = (Class<?>)context.getAttribute(GRANITE_CONFIG_ATTRIBUTE);
075                if (flexFilterClass != null)
076                    context.setAttribute(ServletGraniteConfig.GRANITE_CONFIG_DEFAULT_KEY, "org/granite/config/servlet3/granite-config-servlet3.xml");
077                
078                GraniteConfig gConfig = ServletGraniteConfig.loadConfig(context);
079                ServletServicesConfig.loadConfig(context);
080                
081                if (flexFilterClass != null)
082                    configureServices(context, flexFilterClass);
083                
084                if (gConfig.isRegisterMBeans()) {
085                    GraniteMBeanInitializer.registerMBeans(context, 
086                                    ServletGraniteConfig.getServletConfig(context), 
087                                    ServletServicesConfig.getServletConfig(context));
088                }
089    
090                log.info("GraniteDS initialized");
091            }
092            catch (Exception e) {
093                throw new RuntimeException("Could not initialize Granite environment", e);
094            }
095        }
096    
097        public void contextDestroyed(ServletContextEvent sce) {
098            ServletContext context = sce.getServletContext();
099    
100            log.info("Stopping GraniteDS...");
101    
102            @SuppressWarnings("unchecked")
103            List<ShutdownListener> listeners = (List<ShutdownListener>)sce.getServletContext().getAttribute(GRANITE_CONFIG_SHUTDOWN_KEY);
104            if (listeners != null) {
105                try {
106                    for (ShutdownListener listener : listeners)
107                        listener.stop();
108                }
109                catch (Exception e) {
110                    throw new RuntimeException("Could not destroy Granite environment", e);
111                }
112            }
113    
114            if (ServletParams.get(context, GRANITE_MBEANS_ATTRIBUTE, Boolean.TYPE, false))
115                    GraniteMBeanInitializer.unregisterMBeans(context);
116    
117            log.info("GraniteDS stopped");
118        }
119    
120        public static synchronized void registerShutdownListener(ServletContext context, ShutdownListener listener) {
121            @SuppressWarnings("unchecked")
122            List<ShutdownListener> listeners = (List<ShutdownListener>)context.getAttribute(GRANITE_CONFIG_SHUTDOWN_KEY);
123            if (listeners == null) {
124                listeners = new ArrayList<ShutdownListener>();
125                context.setAttribute(GRANITE_CONFIG_SHUTDOWN_KEY, listeners);
126            }
127            listeners.add(listener);
128        }

131        private void configureServices(ServletContext context, Class<?> flexFilterClass) throws ServletException {
132            GraniteConfig graniteConfig = ServletGraniteConfig.loadConfig(context);
133            ServicesConfig servicesConfig = ServletServicesConfig.loadConfig(context);
134            
135            FlexFilter flexFilter = flexFilterClass.getAnnotation(FlexFilter.class);
136            
137            for (Class<?> ti : flexFilter.tideInterfaces()) {
138                    try {
139                            graniteConfig.getTideComponentMatchers().add(new TideComponentInstanceOfMatcher(ti.getName(), false));
140                            log.debug("Enabled components implementing %s for Tide remoting", ti);
141                    }
142                    catch (Exception e) {
143                            log.error(e, "Could not add tide-component interface %s", ti);
144                    }
145            }
146            for (Class<? extends Annotation> ta : flexFilter.tideAnnotations()) {
147                    try {
148                            graniteConfig.getTideComponentMatchers().add(new TideComponentAnnotatedWithMatcher(ta.getName(), false));
149                            log.debug("Enabled components annotated with %s for Tide remoting", ta);
150                    }
151                    catch (Exception e) {
152                            log.error(e, "Could not add tide-component annotation %s", ta);
153                    }
154            }
155            for (String tn : flexFilter.tideNames()) {
156                    try {
157                            graniteConfig.getTideComponentMatchers().add(new TideComponentNameMatcher(tn, false));
158                            log.debug("Enabled components with name %s for Tide remoting", tn);
159                    }
160                    catch (Exception e) {
161                            log.error(e, "Could not add tide-component name %s", tn);
162                    }
163            }
164            for (String tt : flexFilter.tideTypes()) {
165                    try {
166                            graniteConfig.getTideComponentMatchers().add(new TideComponentTypeMatcher(tt, false));
167                            log.debug("Enabled components with type %s for Tide remoting", tt);
168                    }
169                    catch (Exception e) {
170                            log.error(e, "Could not add tide-component type %s", tt);
171                    }
172            }
173            
174            for (Class<? extends ExceptionConverter> ec : flexFilter.exceptionConverters()) {
175                    graniteConfig.registerExceptionConverter(ec);
176                            log.debug("Registered exception converter %s", ec);
177            }
178            
179            if (!flexFilter.securityServiceClass().equals(SecurityService.class)) {
180                    try {
181                            graniteConfig.setSecurityService(ClassUtil.newInstance(flexFilter.securityServiceClass(), SecurityService.class));
182                    }
183                    catch (Exception e) {
184                            throw new ServletException("Could not setup security service", e);
185                    }
186            }
187            else if (graniteConfig.getSecurityService() == null) {
188                    String securityServiceClass = null;
189                    // Try auto-detect
190                    try {
191                            ClassUtil.forName("org.mortbay.jetty.Request");
192                            securityServiceClass = "org.granite.messaging.service.security.Jetty6SecurityService";
193                    }
194                    catch (ClassNotFoundException e1) {
195                            try {
196                                    ClassUtil.forName("weblogic.servlet.security.ServletAuthentication");
197                                    securityServiceClass = "org.granite.messaging.service.security.WebLogicSecurityService";
198                            }
199                            catch (ClassNotFoundException e2) {
200                                    try {
201                                            ClassUtil.forName("weblogic.servlet.security.ServletAuthentication");
202                                    securityServiceClass = "org.granite.messaging.service.security.WebLogicSecurityService";
203                                    }
204                                    catch (ClassNotFoundException e3) {
205                                    securityServiceClass = "org.granite.messaging.service.security.Tomcat7SecurityService";
206                                    }
207                            }
208                            try {
209                                    graniteConfig.setSecurityService(ClassUtil.newInstance(securityServiceClass, SecurityService.class));
210                    }
211                    catch (Exception e) {
212                            throw new ServletException("Could not setup security service", e);
213                    }
214                    }
215            }
216            
217            if (!flexFilter.amf3MessageInterceptor().equals(AMF3MessageInterceptor.class)) {
218                    try {
219                            graniteConfig.setAmf3MessageInterceptor(ClassUtil.newInstance(flexFilter.amf3MessageInterceptor(), AMF3MessageInterceptor.class));
220                    }
221                    catch (Exception e) {
222                            throw new ServletException("Could not setup amf3 message interceptor", e);
223                    }
224            }
225            
226            Channel channel = servicesConfig.findChannelById("graniteamf");
227            if (channel == null) {
228                    channel = new Channel("graniteamf", "mx.messaging.channels.AMFChannel", 
229                            new EndPoint("http://{server.name}:{server.port}/{context.root}/graniteamf/amf", "flex.messaging.endpoints.AMFEndpoint"), 
230                            new XMap());
231                    servicesConfig.addChannel(channel);
232            }
233    
234            XMap factoryProperties = new XMap();
235            String lookup = "java:global/{context.root}/{capitalized.component.name}Bean";
236            if (context.getClass().getClassLoader().getClass().getName().startsWith("org.jboss"))
237                    lookup = "{capitalized.component.name}Bean/local";
238            if (!("".equals(flexFilter.ejbLookup())))
239                    lookup = flexFilter.ejbLookup();
240            if (lookup.indexOf("{context.root}") >= 0) {
241                    try {
242                            // Call by reflection because of JDK 1.4
243                            Method m = context.getClass().getMethod("getContextPath");
244                            String contextPath = (String)m.invoke(context);
245                            lookup = lookup.replace("{context.root}", contextPath.substring(1));
246                    }
247                    catch (Exception e) {
248                            log.error(e, "Could not get context path, please define lookup manually in @FlexFilter");
249                    }
250            }
251            factoryProperties.put("lookup", lookup);
252    
253            if (flexFilter.tide()) {
254                    Factory factory = servicesConfig.findFactoryById("tide-" + flexFilter.type() + "-factory");
255                    if (factory == null) {
256                            factory = new Factory("tide-" + flexFilter.type() + "-factory", 
257                                    flexFilter.factoryClass().getName(), factoryProperties);
258                            servicesConfig.addFactory(factory);
259                    }
260                    
261                    Service service = servicesConfig.findServiceById("granite-service");
262                    if (service == null) {
263                            service = new Service("granite-service", "flex.messaging.services.RemotingService", 
264                                            "flex.messaging.messages.RemotingMessage", null, null, new HashMap<String, Destination>());
265                            List<String> channelIds = new ArrayList<String>();
266                            channelIds.add("graniteamf");
267                            List<String> tideRoles = flexFilter.tideRoles().length == 0 ? null : Arrays.asList(flexFilter.tideRoles());
268                            Destination destination = new Destination(flexFilter.type(), channelIds, new XMap(), tideRoles, null, null);
269                            destination.getProperties().put("factory", "tide-" + flexFilter.type() + "-factory");
270                            if (!("".equals(flexFilter.entityManagerFactoryJndiName())))
271                                    destination.getProperties().put("entity-manager-factory-jndi-name", flexFilter.entityManagerFactoryJndiName());
272                            else if (!("".equals(flexFilter.entityManagerJndiName())))
273                                    destination.getProperties().put("entity-manager-jndi-name", flexFilter.entityManagerJndiName());
274                            if (!("".equals(flexFilter.validatorClassName())))
275                                    destination.getProperties().put("validator-class-name", flexFilter.validatorClassName());
276                            service.getDestinations().put(flexFilter.type(), destination);
277                            servicesConfig.addService(service);
278                    }
279                
280                    if (flexFilter.factoryClass().getName().equals("org.granite.tide.ejb.EjbServiceFactory"))
281                            servicesConfig.scan(null);
282                    
283                    log.info("Registered Tide service factory and destination");
284            }
285            else {
286                    Factory factory = new Factory(flexFilter.type() + "-factory", flexFilter.factoryClass().getName(), factoryProperties);
287                    servicesConfig.addFactory(factory);
288                    
289                    Service service = new Service("granite-service", "flex.messaging.services.RemotingService", 
290                            "flex.messaging.messages.RemotingMessage", null, null, new HashMap<String, Destination>());
291                    servicesConfig.addService(service);
292                
293                servicesConfig.scan(null);
294                    
295                    log.info("Registered " + flexFilter.factoryClass() + " service factory");
296            }
297        }
