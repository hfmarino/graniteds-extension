157        public GraniteConfig(String stdConfig, InputStream customConfigIs, Configuration configuration, String MBeanContextName) throws IOException, SAXException {
158            try {
159                amf3SerializerConstructor = ClassUtil.getConstructor(AMF3Serializer.class, new Class<?>[]{OutputStream.class});
160                amf3DeserializerConstructor = ClassUtil.getConstructor(AMF3Deserializer.class, new Class<?>[]{InputStream.class});
161            } catch (Exception e) {
162                throw new GraniteConfigException("Could not get constructor for AMF3 (de)serializers", e);
163            }
164            
165            this.MBeanContextName = MBeanContextName;
166            
167            ClassLoader loader = GraniteConfig.class.getClassLoader();
168            
169            final ByteArrayInputStream dtd = StreamUtil.getResourceAsStream("org/granite/config/granite-config.dtd", loader);
170            final EntityResolver resolver = new EntityResolver() {
171                public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
172                    if (GRANITE_CONFIG_PUBLIC_ID.equals(publicId)) {
173                        dtd.reset();
174                        InputSource source = new InputSource(dtd);
175                        source.setPublicId(publicId);
176                        return source;
177                    }
178                    return null;
179                }
180            };
181    
182            // Load standard config.
183            InputStream is = null;
184            try {
185                is = StreamUtil.getResourceAsStream("org/granite/config/granite-config.xml", loader);
186                XMap doc = new XMap(is, resolver);
187                forElement(doc, false, null);
188            } finally {
189                if (is != null)
190                    is.close();
191            }
192            
193            if (stdConfig != null) {
194                try {
195                    is = StreamUtil.getResourceAsStream(stdConfig, loader);
196                    XMap doc = new XMap(is, resolver);
197                    forElement(doc, false, null);
198                } finally {
199                    if (is != null)
200                        is.close();
201                }
202            }
203    
204            // Load custom config (override).
205            if (customConfigIs != null) {
206                    XMap doc = new XMap(customConfigIs, resolver);
207                forElement(doc, true, configuration != null ? configuration.getGraniteConfigProperties() : null);
208            }
209        }

215        private void scanConfig(String graniteConfigProperties) {
216            //if config overriding exists
217            Scanner scanner = ScannerFactory.createScanner(this, graniteConfigProperties != null ? graniteConfigProperties : GRANITE_CONFIG_PROPERTIES);
218            try {
219                scanner.scan();
220            } catch (Exception e) {
221                log.error(e, "Could not scan classpath for configuration");
222            }
223        }

225        public boolean handleMarkerItem(ScannedItem item) {
226            try {
227                return handleProperties(item.loadAsProperties());
228            } catch (Exception e) {
229                log.error(e, "Could not load properties: %s", item);
230            }
231            return true;
232        }
233    
234        public void handleScannedItem(ScannedItem item) {
235            if ("class".equals(item.getExtension()) && item.getName().indexOf('$') == -1) {
236                try {
237                    handleClass(item.loadAsClass());
238                } catch (NoClassDefFoundError e) {
239                    // Ignore errors with Tide classes depending on Gravity
240                } catch (Throwable t) {
241                    log.error(t, "Could not load class: %s", item);
242                }
243            }
244        }
245    
246        private boolean handleProperties(Properties properties) {
247            if (properties.getProperty("dependsOn") != null) {
248                    String dependsOn = properties.getProperty("dependsOn");
249                    try {
250                            ClassUtil.forName(dependsOn);
251                    }
252                    catch (ClassNotFoundException e) {
253                            // Class not found, skip scan for this package
254                            return true;
255                    }
256            }
257            
258            String classGetterName = properties.getProperty("classGetter");
259            if (!classGetterSet && classGetterName != null) {
260                try {
261                    classGetter = ClassUtil.newInstance(classGetterName, ClassGetter.class);
262                } catch (Throwable t) {
263                    log.error(t, "Could not create instance of: %s", classGetterName);
264                }
265            }
266    
267            String amf3MessageInterceptorName = properties.getProperty("amf3MessageInterceptor");
268            if (amf3MessageInterceptor == null && amf3MessageInterceptorName != null) {
269                try {
270                    amf3MessageInterceptor = ClassUtil.newInstance(amf3MessageInterceptorName, AMF3MessageInterceptor.class);
271                } catch (Throwable t) {
272                    log.error(t, "Could not create instance of: %s", amf3MessageInterceptorName);
273                }
274            }
275            
276            for (Map.Entry<?, ?> me : properties.entrySet()) {
277                if (me.getKey().toString().startsWith("converter.")) {
278                    String converterName = me.getValue().toString();
279                    try {
280                        converterClasses.add(ClassUtil.forName(converterName, Converter.class));
281                    } catch (Exception e) {
282                        throw new GraniteConfigException("Could not get converter class for: " + converterName, e);
283                    }
284                }
285            }
286            
287            return false;
288        }
289    
290        private void handleClass(Class<?> clazz) {
291            if (!clazz.isInterface() && !Modifier.isAbstract(clazz.getModifiers())) {
292                if (Externalizer.class.isAssignableFrom(clazz)) {
293                    try {
294                        scannedExternalizers.add(ClassUtil.newInstance(clazz, Externalizer.class));
295                    } catch (Exception e) {
296                        log.error(e, "Could not create new instance of: %s", clazz);
297                    }
298                }
299                
300                if (ExceptionConverter.class.isAssignableFrom(clazz)) {
301                    try {
302                        exceptionConverters.add(ClassUtil.newInstance(clazz, ExceptionConverter.class));
303                    } catch (Exception e) {
304                            if (!clazz.getName().equals("org.granite.tide.hibernate.HibernateValidatorExceptionConverter")) // GDS-582
305                                    log.error(e, "Could not create new instance of: %s", clazz);
306                    }
307                }
308            }
309        }

515        private void forElement(XMap element, boolean custom, String graniteConfigProperties) {
516            String scan = element.get("@scan");
517    
518            this.scan = Boolean.TRUE.toString().equals(scan);
519    
520            loadCustomAMF3Serializer(element, custom);
521            loadCustomAMF3MessageInterceptor(element, custom);
522            loadCustomConverters(element, custom);
523            loadCustomMethodMatcher(element, custom);
524            loadCustomInvocationListener(element, custom);
525            loadCustomInstantiators(element, custom);
526            loadCustomClassGetter(element, custom);
527            loadCustomExternalizers(element, custom);
528            loadCustomDescriptors(element, custom);
529            loadCustomExceptionConverters(element, custom);
530            loadCustomTideComponents(element, custom);
531            loadCustomSecurity(element, custom);
532            loadCustomMessageSelector(element, custom);
533            loadCustomGravity(element, custom);
534    
535            if (this.scan)
536                scanConfig(graniteConfigProperties);
537    
538            finishCustomConverters(custom);
539        }

541        private void loadCustomAMF3Serializer(XMap element, boolean custom) {
542            XMap amf3Serializer = element.getOne("amf3-serializer");
543            if (amf3Serializer != null) {
544                String type = amf3Serializer.get("@type");
545                try {
546                    Class<AMF3Serializer> amf3SerializerClass = ClassUtil.forName(type, AMF3Serializer.class);
547                    amf3SerializerConstructor = ClassUtil.getConstructor(amf3SerializerClass, new Class<?>[]{OutputStream.class});
548                } catch (Exception e) {
549                    throw new GraniteConfigException("Could not get constructor for AMF3 serializer: " + type, e);
550                }
551            }
552    
553            XMap amf3Deserializer = element.getOne("amf3-deserializer");
554            if (amf3Deserializer != null) {
555                String type = amf3Deserializer.get("@type");
556                try {
557                    Class<AMF3Deserializer> amf3DeserializerClass = ClassUtil.forName(type, AMF3Deserializer.class);
558                    amf3DeserializerConstructor = ClassUtil.getConstructor(amf3DeserializerClass, new Class<?>[]{InputStream.class});
559                } catch (Exception e) {
560                    throw new GraniteConfigException("Could not get constructor for AMF3 deserializer: " + type, e);
561                }
562            }
563        }
564    
565        private void loadCustomAMF3MessageInterceptor(XMap element, boolean custom) {
566            XMap methodMatcher = element.getOne("amf3-message-interceptor");
567            if (methodMatcher != null) {
568                String type = methodMatcher.get("@type");
569                try {
570                    amf3MessageInterceptor = (AMF3MessageInterceptor)ClassUtil.newInstance(type);
571                } catch (Exception e) {
572                    throw new GraniteConfigException("Could not construct amf3 message interceptor: " + type, e);
573                }
574            }
575        }
576    
577        private void loadCustomConverters(XMap element, boolean custom) {
578            XMap converters = element.getOne("converters");
579            if (converters != null) {
580                // Should we override standard config converters?
581                String override = converters.get("@override");
582                if (Boolean.TRUE.toString().equals(override))
583                    converterClasses.clear();
584    
585                int i = 0;
586                for (XMap converter : converters.getAll("converter")) {
587                    String type = converter.get("@type");
588                    try {
589                        // For custom config, shifts any standard converters to the end of the list...
590                        converterClasses.add(i++, ClassUtil.forName(type, Converter.class));
591                    } catch (Exception e) {
592                        throw new GraniteConfigException("Could not get converter class for: " + type, e);
593                    }
594                }
595            }
596        }
597        
598        private void finishCustomConverters(boolean custom) {
599            try {
600                converters = new Converters(converterClasses);
601            } catch (Exception e) {
602                throw new GraniteConfigException("Could not construct new Converters instance", e);
603            }
604            
605            // Cleanup...
606            if (custom)
607                converterClasses = null;
608        }

655        private void loadCustomExternalizers(XMap element, boolean custom) {
656            externalizersConfiguration = element.getOne("externalizers/configuration");
657            
658            for (XMap externalizer : element.getAll("externalizers/externalizer")) {
659                String externalizerType = externalizer.get("@type");
660    
661                for (XMap include : externalizer.getAll("include")) {
662                    String type = include.get("@type");
663                    if (type != null)
664                        externalizersByType.put(type, EXTERNALIZER_FACTORY.getInstance(externalizerType, this));
665                    else {
666                        String instanceOf = include.get("@instance-of");
667                        if (instanceOf != null)
668                            externalizersByInstanceOf.put(instanceOf, externalizerType);
669                        else {
670                            String annotatedWith = include.get("@annotated-with");
671                            if (annotatedWith == null)
672                                throw new GraniteConfigException(
673                                    "Element 'include' has no attribute 'type', 'instance-of' or 'annotated-with'");
674                            externalizersByAnnotatedWith.put(annotatedWith, externalizerType);
675                        }
676                    }
677                }
678            }
679        }

681        /**
682         * Read custom class descriptors.
683         * Descriptor must have 'type' or 'instanceof' attribute
684         * and one of 'java' or 'as3' attributes specified.
685         */
686        private void loadCustomDescriptors(XMap element, boolean custom) {
687            for (XMap descriptor : element.getAll("descriptors/descriptor")) {
688                String type = descriptor.get("@type");
689                if (type != null) {
690                    String java = descriptor.get("@java");
691                    String as3 = descriptor.get("@as3");
692                    if (java == null && as3 == null)
693                        throw new GraniteConfigException(
694                            "Element 'descriptor' has no attributes 'java' or 'as3'\n" + descriptor
695                        );
696                    if (java != null)
697                        javaDescriptorsByType.put(type, JC_DESCRIPTOR_FACTORY.getInstance(java, this));
698                    if (as3 != null)
699                        as3DescriptorsByType.put(type, ASC_DESCRIPTOR_FACTORY.getInstance(as3, this));
700                } else {
701                    String instanceOf = descriptor.get("@instance-of");
702                    if (instanceOf == null)
703                        throw new GraniteConfigException(
704                            "Element 'descriptor' has no attribute 'type' or 'instance-of'\n" + descriptor
705                        );
706                    String java = descriptor.get("@java");
707                    String as3 = descriptor.get("@as3");
708                    if (java == null && as3 == null) {
709                        throw new GraniteConfigException(
710                            "Element 'descriptor' has no attributes 'java' or 'as3' in:\n" + descriptor
711                        );
712                    }
713                    if (java != null)
714                        javaDescriptorsByInstanceOf.put(instanceOf, java);
715                    if (as3 != null)
716                        as3DescriptorsByInstanceOf.put(instanceOf, as3);
717                }
718            }
719        }

785            }

811        private <T> T getElementByType(
812            String type,
813            ConfigurableFactory<T> factory,
814            ConcurrentHashMap<String, T> elementsByType,
815            Map<String, String> elementsByInstanceOf,
816            Map<String, String> elementsByAnnotatedWith,
817            List<T> scannedConfigurables) {
818    
819            // This NULL object is a Java null placeholder: ConcurrentHashMap doesn't allow
820            // null values...
821            final T NULL = factory.getNullInstance();
822    
823            T element = elementsByType.get(type);
824            if (element != null)
825                return (NULL == element ? null : element);
826            element = NULL;
827    
828            Class<?> typeClass = null;
829            try {
830                typeClass = ClassUtil.forName(type);
831            } catch (Exception e) {
832                throw new GraniteConfigException("Could not load class: " + type, e);
833            }
834    
835            if (elementsByAnnotatedWith != null && NULL == element) {
836                for (Map.Entry<String, String> entry : elementsByAnnotatedWith.entrySet()) {
837                    String annotation = entry.getKey();
838                    try {
839                        Class<Annotation> annotationClass = ClassUtil.forName(annotation, Annotation.class);
840                        if (typeClass.isAnnotationPresent(annotationClass)) {
841                            element = factory.getInstance(entry.getValue(), this);
842                            break;
843                        }
844                    } catch (Exception e) {
845                        throw new GraniteConfigException("Could not load class: " + annotation, e);
846                    }
847                }
848            }
849    
850            if (elementsByInstanceOf != null && NULL == element) {
851                    for (Map.Entry<String, String> entry : elementsByInstanceOf.entrySet()) {
852                        String instanceOf = entry.getKey();
853                        try {
854                            Class<?> instanceOfClass = ClassUtil.forName(instanceOf);
855                            if (instanceOfClass.isAssignableFrom(typeClass)) {
856                                element = factory.getInstance(entry.getValue(), this);
857                                break;
858                            }
859                        } catch (Exception e) {
860                            throw new GraniteConfigException("Could not load class: " + instanceOf, e);
861                        }
862                    }
863            }
864    
865            if (NULL == element)
866                element = factory.getInstanceForBean(scannedConfigurables, typeClass, this);
867    
868            T previous = elementsByType.putIfAbsent(type, element);
869            if (previous != null)
870                element = previous;
871    
872            return (NULL == element ? null : element);
873        }
