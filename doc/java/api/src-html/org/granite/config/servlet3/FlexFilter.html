023    import static java.lang.annotation.ElementType.TYPE;
024    import static java.lang.annotation.RetentionPolicy.RUNTIME;
025    
026    import java.lang.annotation.Annotation;
027    import java.lang.annotation.Inherited;
028    import java.lang.annotation.Retention;
029    import java.lang.annotation.Target;
030    
031    import org.granite.messaging.amf.process.AMF3MessageInterceptor;
032    import org.granite.messaging.service.ExceptionConverter;
033    import org.granite.messaging.service.ServiceFactory;
034    import org.granite.messaging.service.annotations.RemoteDestination;
035    import org.granite.messaging.service.security.SecurityService;
036    
037    
038    @Inherited
039    @Retention(RUNTIME)
040    @Target({TYPE})
041    public @interface FlexFilter {

043            String graniteUrlMapping() default "/graniteamf/*";

045            String gravityUrlMapping() default "/gravityamf/*";

047            boolean tide() default false;

049            String type();
050            
051            Class<? extends ServiceFactory> factoryClass();
052            
053            Class<? extends SecurityService> securityServiceClass() default SecurityService.class;
054            
055            Class<?>[] tideInterfaces() default {};
056            
057            Class<? extends Annotation>[] tideAnnotations() default { RemoteDestination.class };
058            
059            String[] tideNames() default {};
060            
061            String[] tideTypes() default {};
062            
063            String[] tideRoles() default {};
064            
065            Class<? extends ExceptionConverter>[] exceptionConverters() default {};
066            
067            Class<? extends AMF3MessageInterceptor> amf3MessageInterceptor() default AMF3MessageInterceptor.class;
068            
069            String ejbLookup() default "";
070            
071            String entityManagerFactoryJndiName() default "";
072            
073            String entityManagerJndiName() default "";
074            
075            String validatorClassName() default "";
076    }
