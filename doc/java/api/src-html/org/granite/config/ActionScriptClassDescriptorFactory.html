021    package org.granite.config;
022    
023    import java.util.List;
024    
025    import org.granite.messaging.amf.io.util.ActionScriptClassDescriptor;
026    import org.granite.util.ClassUtil;
027    
028    /**
029     * @author Franck WOLFF
030     */
031    public class ActionScriptClassDescriptorFactory implements ConfigurableFactory<Class<? extends ActionScriptClassDescriptor>> {
032    
033        private static final class NullActionScriptClassDescriptor extends ActionScriptClassDescriptor {
034            public NullActionScriptClassDescriptor(String type, byte encoding) {
035                super(type, encoding);
036                throw new RuntimeException("Not implemented");
037            }
038            @Override
039            public void defineProperty(String name) {
040            }
041            @Override
042            public Object newJavaInstance() {
043                return null;
044            }
045        }
046        private static final Class<? extends ActionScriptClassDescriptor> NULL_ASC_DESCRIPTOR
047            = NullActionScriptClassDescriptor.class;
048    
049        public Class<? extends ActionScriptClassDescriptor> getNullInstance() {
050            return NULL_ASC_DESCRIPTOR;
051        }
052    
053        public Class<? extends ActionScriptClassDescriptor> getInstance(String type, GraniteConfig config) throws GraniteConfigException {
054            try {
055                return ClassUtil.forName(type, ActionScriptClassDescriptor.class);
056            } catch (Exception e) {
057                throw new GraniteConfigException("Could not load ActionScriptClassDescriptor type: " + type, e);
058            }
059        }
060    
061        public Class<? extends ActionScriptClassDescriptor> getInstanceForBean(
062            List<Class<? extends ActionScriptClassDescriptor>> scannedConfigurables,
063            Class<?> beanClass,
064            GraniteConfig config) throws GraniteConfigException {
065    
066            return NULL_ASC_DESCRIPTOR;
067        }
068    }
