083        public Object readObject() throws IOException {
084            if (debugMore) logMore.debug("readObject()...");
085    
086            try {
087                    int type = readAMF3Integer();
088                    return readObject(type);
089            }
090            catch (IOException e) {
091                    throw e;
092            }
093            catch (Exception e) {
094                    throw new AMF3SerializationException(e);
095            }
096        }

101        protected Object readObject(int type) throws IOException {
102    
103            if (debugMore) logMore.debug("readObject(type=0x%02X)", type);
104    
105            switch (type) {
106            case AMF3_UNDEFINED: // 0x00;
107            case AMF3_NULL: // 0x01;
108                return null;
109            case AMF3_BOOLEAN_FALSE: // 0x02;
110                return Boolean.FALSE;
111            case AMF3_BOOLEAN_TRUE: // 0x03;
112                return Boolean.TRUE;
113            case AMF3_INTEGER: // 0x04;
114                return Integer.valueOf(readAMF3Integer());
115            case AMF3_NUMBER: // 0x05;
116                return readAMF3Double();
117            case AMF3_STRING: // 0x06;
118                return readAMF3String();
119            case AMF3_XML: // 0x07;
120                return readAMF3Xml();
121            case AMF3_DATE: // 0x08;
122                return readAMF3Date();
123            case AMF3_ARRAY: // 0x09;
124                return readAMF3Array();
125            case AMF3_OBJECT: // 0x0A;
126                return readAMF3Object();
127            case AMF3_XMLSTRING: // 0x0B;
128                return readAMF3XmlString();
129            case AMF3_BYTEARRAY: // 0x0C;
130                return readAMF3ByteArray();
131    
132            case AMF3_VECTOR_INT: // 0x0D;
133                    return readAMF3VectorInt();
134            case AMF3_VECTOR_UINT: // 0x0E;
135                    return readAMF3VectorUInt();
136            case AMF3_VECTOR_NUMBER: // 0x0F;
137                    return readAMF3VectorNumber();
138            case AMF3_VECTOR_OBJECT: // 0x10;
139                    return readAMF3VectorObject();
140    
141            default:
142                throw new IllegalArgumentException("Unknown type: " + type);
143            }
144        }

146        protected int readAMF3Integer() throws IOException {
147            int result = 0;
148    
149            int n = 0;
150            int b = readUnsignedByte();
151            while ((b & 0x80) != 0 && n < 3) {
152                result <<= 7;
153                result |= (b & 0x7f);
154                b = readUnsignedByte();
155                n++;
156            }
157            if (n < 3) {
158                result <<= 7;
159                result |= b;
160            } else {
161                result <<= 8;
162                result |= b;
163                if ((result & 0x10000000) != 0)
164                    result |= 0xe0000000;
165            }
166    
167            if (debugMore) logMore.debug("readAMF3Integer() -> %d", result);
168    
169            return result;
170        }

172        protected Double readAMF3Double() throws IOException {
173            double d = readDouble();
174            Double result = (Double.isNaN(d) ? null : Double.valueOf(d));
175    
176            if (debugMore) logMore.debug("readAMF3Double() -> %f", result);
177    
178            return result;
179        }

181        protected String readAMF3String() throws IOException {
182            String result = null;
183    
184            if (debugMore) logMore.debug("readAMF3String()...");
185    
186            int type = readAMF3Integer();
187            if ((type & 0x01) == 0) // stored string
188                result = getFromStoredStrings(type >> 1);
189            else {
190                int length = type >> 1;
191                if (debugMore) logMore.debug("readAMF3String() - length=%d", length);
192    
193                if (length > 0) {
194    
195                    byte[] utfBytes = new byte[length];
196                    char[] utfChars = new char[length];
197    
198                    readFully(utfBytes);
199    
200                    int c, c2, c3, iBytes = 0, iChars = 0;
201                    while (iBytes < length) {
202                        c = utfBytes[iBytes++] & 0xFF;
203                        if (c <= 0x7F)
204                            utfChars[iChars++] = (char)c;
205                        else {
206                            switch (c >> 4) {
207                            case 12: case 13:
208                                c2 = utfBytes[iBytes++];
209                                if ((c2 & 0xC0) != 0x80)
210                                    throw new UTFDataFormatException("Malformed input around byte " + (iBytes-2));
211                                utfChars[iChars++] = (char)(((c & 0x1F) << 6) | (c2 & 0x3F));
212                                break;
213                            case 14:
214                                c2 = utfBytes[iBytes++];
215                                c3 = utfBytes[iBytes++];
216                                if (((c2 & 0xC0) != 0x80) || ((c3 & 0xC0) != 0x80))
217                                    throw new UTFDataFormatException("Malformed input around byte " + (iBytes-3));
218                                utfChars[iChars++] = (char)(((c & 0x0F) << 12) | ((c2 & 0x3F) << 6) | ((c3 & 0x3F) << 0));
219                                break;
220                            default:
221                                throw new UTFDataFormatException("Malformed input around byte " + (iBytes-1));
222                            }
223                        }
224                    }
225                    result = new String(utfChars, 0, iChars);
226    
227                    if (debugMore) logMore.debug("readAMF3String() - result=%s", result);
228    
229                    addToStoredStrings(result);
230                } else
231                    result = "";
232            }
233    
234            if (debugMore) logMore.debug("readAMF3String() -> %s", result);
235    
236            return result;
237        }

240        protected Date readAMF3Date() throws IOException {
241            Date result = null;
242    
243            int type = readAMF3Integer();
244            if ((type & 0x01) == 0) // stored Date
245                result = (Date)getFromStoredObjects(type >> 1);
246            else {
247                result = new Date((long)readDouble());
248                addToStoredObjects(result);
249            }
250    
251            if (debugMore) logMore.debug("readAMF3Date() -> %s", result);
252    
253            return result;
254        }

256        protected Object readAMF3Array() throws IOException {
257            Object result = null;
258    
259            int type = readAMF3Integer();
260            if ((type & 0x01) == 0) // stored array.
261                result = getFromStoredObjects(type >> 1);
262            else {
263                final int size = type >> 1;
264    
265                String key = readAMF3String();
266                if (key.length() == 0) {
267                    Object[] objects = new Object[size];
268                    addToStoredObjects(objects);
269    
270                    for (int i = 0; i < size; i++)
271                        objects[i] = readObject();
272    
273                    result = objects;
274                }
275                else {
276                    Map<Object, Object> map = new HashMap<Object, Object>();
277                    addToStoredObjects(map);
278    
279                    while(key.length() > 0) {
280                        map.put(key, readObject());
281                        key = readAMF3String();
282                    }
283                    for (int i = 0; i < size; i++)
284                        map.put(Integer.valueOf(i), readObject());
285    
286                    result = map;
287                }
288            }
289    
290            if (debugMore) logMore.debug("readAMF3Array() -> %s", result);
291    
292            return result;
293        }

396        protected Object readAMF3Object() throws IOException {
397            if (debug) log.debug("readAMF3Object()...");
398    
399            Object result = null;
400    
401            int type = readAMF3Integer();
402            if (debug) log.debug("readAMF3Object() - type=0x%02X", type);
403    
404            if ((type & 0x01) == 0) // stored object.
405                result = getFromStoredObjects(type >> 1);
406            else {
407                boolean inlineClassDef = (((type >> 1) & 0x01) != 0);
408                if (debug) log.debug("readAMF3Object() - inlineClassDef=%b", inlineClassDef);
409    
410                // read class decriptor.
411                ActionScriptClassDescriptor desc = null;
412                if (inlineClassDef) {
413                    int propertiesCount = type >> 4;
414                    if (debug) log.debug("readAMF3Object() - propertiesCount=%d", propertiesCount);
415    
416                    byte encoding = (byte)((type >> 2) & 0x03);
417                    if (debug) log.debug("readAMF3Object() - encoding=%d", encoding);
418    
419                    String className = readAMF3String();
420                    if (debug) log.debug("readAMF3Object() - className=%s", className);
421    
422                    // try to find out custom AS3 class descriptor
423                    Class<? extends ActionScriptClassDescriptor> descriptorType = null;
424                    if (!"".equals(className))
425                        descriptorType = context.getGraniteConfig().getActionScriptDescriptor(className);
426                    if (debug) log.debug("readAMF3Object() - descriptorType=%s", descriptorType);
427    
428                    if (descriptorType != null) {
429                        // instantiate descriptor
430                        Class<?>[] argsDef = new Class[]{String.class, byte.class};
431                        Object[] argsVal = new Object[]{className, Byte.valueOf(encoding)};
432                        try {
433                            desc = ClassUtil.newInstance(descriptorType, argsDef, argsVal);
434                        } catch (Exception e) {
435                            throw new RuntimeException("Could not instantiate AS descriptor: " + descriptorType, e);
436                        }
437                    }
438                    if (desc == null)
439                        desc = new DefaultActionScriptClassDescriptor(className, encoding);
440                    addToStoredClassDescriptors(desc);
441    
442                    if (debug) log.debug("readAMF3Object() - defining %d properties...", propertiesCount);
443                    for (int i = 0; i < propertiesCount; i++) {
444                        String name = readAMF3String();
445                        if (debug) log.debug("readAMF3Object() - defining property name=%s", name);
446                        desc.defineProperty(name);
447                    }
448                } else
449                    desc = getFromStoredClassDescriptors(type >> 2);
450    
451                if (debug) log.debug("readAMF3Object() - actionScriptClassDescriptor=%s", desc);
452    
453                int objectEncoding = desc.getEncoding();
454    
455                // Find externalizer and create Java instance.
456                Externalizer externalizer = desc.getExternalizer();
457                if (externalizer != null) {
458                    try {
459                        result = externalizer.newInstance(desc.getType(), this);
460                    } catch (Exception e) {
461                        throw new RuntimeException("Could not instantiate type: " + desc.getType(), e);
462                    }
463                } else
464                    result = desc.newJavaInstance();
465    
466                int index = addToStoredObjects(result);
467                
468                // Entity externalizers (eg. OpenJPA) may return null values for non-null AS3 objects (ie. proxies).
469                if (result == null) {
470                    if (debug) log.debug("readAMF3Object() - Added null object to stored objects for actionScriptClassDescriptor=%s", desc);
471                    return null;
472                }
473    
474                // read object content...
475                if ((objectEncoding & 0x01) != 0) {
476                    // externalizer.
477                    if (externalizer != null) {
478                        if (debug) log.debug("readAMF3Object() - using externalizer=%s", externalizer);
479                        try {
480                            externalizer.readExternal(result, this);
481                        } catch (IOException e) {
482                            throw e;
483                        } catch (Exception e) {
484                            throw new RuntimeException("Could not read externalized object: " + result, e);
485                        }
486                    }
487                    // legacy externalizable.
488                    else {
489                            if (debug) log.debug("readAMF3Object() - legacy Externalizable=%s", result.getClass());
490                            if (!(result instanceof Externalizable)) {
491                                    throw new RuntimeException(
492                                            "The ActionScript3 class bound to " + result.getClass().getName() +
493                                            " (ie: [RemoteClass(alias=\"" + result.getClass().getName() + "\")])" +
494                                            " implements flash.utils.IExternalizable but this Java class neither" +
495                                            " implements java.io.Externalizable nor is in the scope of a configured" +
496                                            " externalizer (please fix your granite-config.xml)"
497                                    );
498                            }
499                        try {
500                            ((Externalizable)result).readExternal(this);
501                        } catch (IOException e) {
502                            throw e;
503                        } catch (Exception e) {
504                            throw new RuntimeException("Could not read externalizable object: " + result, e);
505                        }
506                    }
507                }
508                else {
509                    // defined values...
510                    if (desc.getPropertiesCount() > 0) {
511                        if (debug) log.debug("readAMF3Object() - reading defined properties...");
512                        for (int i = 0; i < desc.getPropertiesCount(); i++) {
513                            byte vType = readByte();
514                            Object value = readObject(vType);
515                            if (debug) log.debug("readAMF3Object() - setting defined property: %s=%s", desc.getPropertyName(i), value);
516                            desc.setPropertyValue(i, result, value);
517                        }
518                    }
519    
520                    // dynamic values...
521                    if (objectEncoding == 0x02) {
522                        if (debug) log.debug("readAMF3Object() - reading dynamic properties...");
523                        while (true) {
524                            String name = readAMF3String();
525                            if (name.length() == 0)
526                                break;
527                            byte vType = readByte();
528                            Object value = readObject(vType);
529                            if (debug) log.debug("readAMF3Object() - setting dynamic property: %s=%s", name, value);
530                            desc.setPropertyValue(name, result, value);
531                        }
532                    }
533                }
534    
535                if (result instanceof AbstractInstantiator<?>) {
536                    if (debug) log.debug("readAMF3Object() - resolving instantiator...");
537                    try {
538                        result = ((AbstractInstantiator<?>)result).resolve();
539                    } catch (Exception e) {
540                        throw new RuntimeException("Could not instantiate object: " + result, e);
541                    }
542                    setStoredObject(index, result);
543                }
544            }
545    
546            if (debug) log.debug("readAMF3Object() -> %s", result);
547    
548            return result;
549        }

560        protected String readAMF3XmlString() throws IOException {
561            String result = null;
562    
563            int type = readAMF3Integer();
564            if ((type & 0x01) == 0) // stored String
565                result = getFromStoredStrings(type >> 1);
566            else {
567                byte[] bytes = readBytes(type >> 1);
568                result = new String(bytes, "UTF-8");
569                addToStoredStrings(result);
570            }
571    
572            if (debugMore) logMore.debug("readAMF3XmlString() -> %s", result);
573    
574            return result;
575        }

577        protected byte[] readAMF3ByteArray() throws IOException {
578            byte[] result = null;
579    
580            int type = readAMF3Integer();
581            if ((type & 0x01) == 0) // stored object.
582                result = (byte[])getFromStoredObjects(type >> 1);
583            else {
584                result = readBytes(type >> 1);
585                addToStoredObjects(result);
586            }
587    
588            if (debugMore) logMore.debug("readAMF3ByteArray() -> %s", result);
589    
590            return result;
591        }

595    
596        protected void addToStoredStrings(String s) {
597            if (debug) log.debug("addToStoredStrings(s=%s) at index=%d", s, storedStrings.size());
598            storedStrings.add(s);
599        }
600    
601        protected String getFromStoredStrings(int index) {
602            if (debug) log.debug("getFromStoredStrings(index=%d)", index);
603            String s = storedStrings.get(index);
604            if (debug) log.debug("getFromStoredStrings() -> %s", s);
605            return s;
606        }
607    
608        protected int addToStoredObjects(Object o) {
609            int index = storedObjects.size();
610            if (debug) log.debug("addToStoredObjects(o=%s) at index=%d", o, index);
611            storedObjects.add(o);
612            return index;
613        }
614    
615        protected void setStoredObject(int index, Object o) {
616            if (debug) log.debug("setStoredObject(index=%d, o=%s)", index, o);
617            storedObjects.set(index, o);
618        }
619    
620        protected Object getFromStoredObjects(int index) {
621            if (debug) log.debug("getFromStoredObjects(index=%d)", index);
622            Object o = storedObjects.get(index);
623            if (debug) log.debug("getFromStoredObjects() -> %s", o);
624            return o;
625        }
626    
627        protected void addToStoredClassDescriptors(ActionScriptClassDescriptor desc) {
628            if (debug) log.debug("addToStoredClassDescriptors(desc=%s) at index=%d", desc, storedClassDescriptors.size());
629            storedClassDescriptors.add(desc);
630        }
631    
632        protected ActionScriptClassDescriptor getFromStoredClassDescriptors(int index) {
633            if (debug) log.debug("getFromStoredClassDescriptors(index=%d)", index);
634            ActionScriptClassDescriptor desc = storedClassDescriptors.get(index);
635            if (debug) log.debug("getFromStoredClassDescriptors() -> %s", desc);
636            return desc;
637        }

642        protected byte[] readBytes(int count) throws IOException {
643            byte[] bytes = new byte[count];
644            readFully(bytes);
645            return bytes;
646        }
