107        public void writeObject(Object o) throws IOException {
108            if (debugMore) logMore.debug("writeObject(o=%s)", o);
109    
110            try {
111                    if (o == null)
112                        write(AMF3_NULL);
113                    else if (!(o instanceof Externalizable)) {
114            
115                        if (converters.hasReverters())
116                            o = converters.revert(o);
117            
118                        if (o instanceof String || o instanceof Character)
119                            writeAMF3String(o.toString());
120                        else if (o instanceof Boolean)
121                            write(((Boolean)o).booleanValue() ? AMF3_BOOLEAN_TRUE : AMF3_BOOLEAN_FALSE);
122                        else if (o instanceof Number) {
123                            if (o instanceof Integer || o instanceof Short || o instanceof Byte)
124                                writeAMF3Integer(((Number)o).intValue());
125                            else if (externalizeLong && o instanceof Long)
126                                    writeAMF3Object(o);
127                            else if (externalizeBigInteger && o instanceof BigInteger)
128                                    writeAMF3Object(o);
129                            else if (externalizeBigDecimal && o instanceof BigDecimal)
130                                    writeAMF3Object(o);
131                            else
132                                writeAMF3Number(((Number)o).doubleValue());
133                        }
134                        else if (o instanceof Date)
135                            writeAMF3Date((Date)o);
136                        else if (o instanceof Calendar)
137                            writeAMF3Date(((Calendar)o).getTime());
138                        else if (o instanceof Document)
139                            writeAMF3Xml((Document)o);
140                        else if (o instanceof Collection<?>)
141                            writeAMF3Collection((Collection<?>)o);
142                        else if (o.getClass().isArray()) {
143                            if (o.getClass().getComponentType() == Byte.TYPE)
144                                writeAMF3ByteArray((byte[])o);
145                            else
146                                writeAMF3Array(o);
147                        }
148                        else
149                            writeAMF3Object(o);
150                    } else
151                        writeAMF3Object(o);
152            }
153            catch (IOException e) {
154                    throw e;
155            }
156            catch (Exception e) {
157                    throw new AMF3SerializationException(e);
158            }
159        }

164        protected void writeAMF3Integer(int i) throws IOException {
165            if (debugMore) logMore.debug("writeAMF3Integer(i=%d)", i);
166    
167            if (i < AMF3_INTEGER_MIN || i > AMF3_INTEGER_MAX) {
168                if (debugMore) logMore.debug("writeAMF3Integer() - %d is out of AMF3 int range, writing it as a Number", i);
169                writeAMF3Number(i);
170            } else {
171                write(AMF3_INTEGER);
172                writeAMF3IntegerData(i);
173            }
174        }
175    
176        protected void writeAMF3IntegerData(int i) throws IOException {
177            if (debugMore) logMore.debug("writeAMF3IntegerData(i=%d)", i);
178    
179            if (i < AMF3_INTEGER_MIN || i > AMF3_INTEGER_MAX)
180                throw new IllegalArgumentException("Integer out of range: " + i);
181    
182            if (i < 0 || i >= 0x200000) {
183                write(((i >> 22) & 0x7F) | 0x80);
184                write(((i >> 15) & 0x7F) | 0x80);
185                write(((i >> 8) & 0x7F) | 0x80);
186                write(i & 0xFF);
187            } else {
188                if (i >= 0x4000)
189                    write(((i >> 14) & 0x7F) | 0x80);
190                if (i >= 0x80)
191                    write(((i >> 7) & 0x7F) | 0x80);
192                write(i & 0x7F);
193            }
194        }

197            if (debugMore) logMore.debug("writeAMF3Number(d=%f)", d);

203        protected void writeAMF3String(String s) throws IOException {
204            if (debugMore) logMore.debug("writeAMF3String(s=%s)", s);
205    
206            write(AMF3_STRING);
207            writeAMF3StringData(s);
208        }

210        protected void writeAMF3StringData(String s) throws IOException {
211            if (debugMore) logMore.debug("writeAMF3StringData(s=%s)", s);
212    
213            if (s.length() == 0) {
214                write(0x01);
215                return;
216            }
217    
218            int index = indexOfStoredStrings(s);
219    
220            if (index >= 0)
221                writeAMF3IntegerData(index << 1);
222            else {
223                addToStoredStrings(s);
224    
225                final int sLength = s.length();
226    
227                // Compute and write modified UTF-8 string length.
228                int uLength = 0;
229                for (int i = 0; i < sLength; i++) {
230                    int c = s.charAt(i);
231                    if ((c >= 0x0001) && (c <= 0x007F))
232                        uLength++;
233                    else if (c > 0x07FF)
234                        uLength += 3;
235                    else
236                        uLength += 2;
237                }
238                writeAMF3IntegerData((uLength << 1) | 0x01);
239    
240                // Write modified UTF-8 bytes.
241                for (int i = 0; i < sLength; i++) {
242                    int c = s.charAt(i);
243                    if ((c >= 0x0001) && (c <= 0x007F)) {
244                        write(c);
245                    } else if (c > 0x07FF) {
246                        write(0xE0 | ((c >> 12) & 0x0F));
247                        write(0x80 | ((c >>  6) & 0x3F));
248                        write(0x80 | ((c >>  0) & 0x3F));
249                    } else {
250                        write(0xC0 | ((c >>  6) & 0x1F));
251                        write(0x80 | ((c >>  0) & 0x3F));
252                    }
253                }
254            }
255        }

257        protected void writeAMF3Xml(Document doc) throws IOException {
258            if (debugMore) logMore.debug("writeAMF3Xml(doc=%s)", doc);
259    
260            byte xmlType = AMF3_XMLSTRING;
261            Channel channel = getChannel();
262            if (channel != null && channel.isLegacyXmlSerialization())
263                xmlType = AMF3_XML;
264            write(xmlType);
265    
266            int index = indexOfStoredObjects(doc);
267            if (index >= 0)
268                writeAMF3IntegerData(index << 1);
269            else {
270                addToStoredObjects(doc);
271    
272                byte[] bytes = xmlUtil.toString(doc).getBytes("UTF-8");
273                writeAMF3IntegerData((bytes.length << 1) | 0x01);
274                write(bytes);
275            }
276        }

278        protected void writeAMF3Date(Date date) throws IOException {
279            if (debugMore) logMore.debug("writeAMF3Date(date=%s)", date);
280    
281            write(AMF3_DATE);
282    
283            int index = indexOfStoredObjects(date);
284            if (index >= 0)
285                writeAMF3IntegerData(index << 1);
286            else {
287                addToStoredObjects(date);
288                writeAMF3IntegerData(0x01);
289                writeDouble(date.getTime());
290            }

293        protected void writeAMF3Array(Object array) throws IOException {
294            if (debugMore) logMore.debug("writeAMF3Array(array=%s)", array);
295    
296            write(AMF3_ARRAY);
297    
298            int index = indexOfStoredObjects(array);
299            if (index >= 0)
300                writeAMF3IntegerData(index << 1);
301            else {
302                addToStoredObjects(array);
303    
304                int length = Array.getLength(array);
305                writeAMF3IntegerData(length << 1 | 0x01);
306                write(0x01);
307                for (int i = 0; i < length; i++)
308                    writeObject(Array.get(array, i));
309            }
310        }

319                writeAMF3IntegerData(index << 1);
320            else {
321                addToStoredObjects(bytes);
322    
323                writeAMF3IntegerData(bytes.length << 1 | 0x01);
324                write(bytes);
325            }

328        protected void writeAMF3Collection(Collection<?> c) throws IOException {
329            if (debugMore) logMore.debug("writeAMF3Collection(c=%s)", c);
330    
331            Channel channel = getChannel();
332            if (channel != null && channel.isLegacyCollectionSerialization())
333                writeAMF3Array(c.toArray());
334            else {
335                ArrayCollection ac = (c instanceof ArrayCollection ? (ArrayCollection)c : new ArrayCollection(c));
336                writeAMF3Object(ac);
337            }
338        }

340        protected void writeAMF3Object(Object o) throws IOException {
341            if (debug) log.debug("writeAMF3Object(o=%s)...", o);
342    
343            write(AMF3_OBJECT);
344    
345            int index = indexOfStoredObjects(o);
346            if (index >= 0)
347                writeAMF3IntegerData(index << 1);
348            else {
349                addToStoredObjects(o);
350    
351                ClassGetter classGetter = context.getGraniteConfig().getClassGetter();
352                if (debug) log.debug("writeAMF3Object() - classGetter=%s", classGetter);
353    
354                Class<?> oClass = classGetter.getClass(o);
355                if (debug) log.debug("writeAMF3Object() - oClass=%s", oClass);
356    
357                JavaClassDescriptor desc = null;
358    
359                // write class description.
360                IndexedJavaClassDescriptor iDesc = getFromStoredClassDescriptors(oClass);
361                if (iDesc != null) {
362                    desc = iDesc.getDescriptor();
363                    writeAMF3IntegerData(iDesc.getIndex() << 2 | 0x01);
364                }
365                else {
366                    iDesc = addToStoredClassDescriptors(oClass);
367                    desc = iDesc.getDescriptor();
368    
369                    writeAMF3IntegerData((desc.getPropertiesCount() << 4) | (desc.getEncoding() << 2) | 0x03);
370                    writeAMF3StringData(desc.getName());
371    
372                    for (int i = 0; i < desc.getPropertiesCount(); i++)
373                        writeAMF3StringData(desc.getPropertyName(i));
374                }
375                if (debug) log.debug("writeAMF3Object() - desc=%s", desc);
376    
377                // write object content.
378                if (desc.isExternalizable()) {
379                    Externalizer externalizer = desc.getExternalizer();
380    
381                    if (externalizer != null) {
382                        if (debug) log.debug("writeAMF3Object() - using externalizer=%s", externalizer);
383                        try {
384                            externalizer.writeExternal(o, this);
385                        } catch (IOException e) {
386                            throw e;
387                        } catch (Exception e) {
388                            throw new RuntimeException("Could not externalize object: " + o, e);
389                        }
390                    }
391                    else {
392                        if (debug) log.debug("writeAMF3Object() - legacy Externalizable=%s", o);
393                        ((Externalizable)o).writeExternal(this);
394                    }
395                }
396                else {
397                    if (debug) log.debug("writeAMF3Object() - writing defined properties...");
398                    for (int i = 0; i < desc.getPropertiesCount(); i++) {
399                        Object obj = desc.getPropertyValue(i, o);
400                        if (debug) log.debug("writeAMF3Object() - writing defined property: %s=%s", desc.getPropertyName(i), obj);
401                        writeObject(obj);
402                    }
403    
404                    if (desc.isDynamic()) {
405                        if (debug) log.debug("writeAMF3Object() - writing dynamic properties...");
406                        Map<?, ?> oMap = (Map<?, ?>)o;
407                        for (Map.Entry<?, ?> entry : oMap.entrySet()) {
408                            Object key = entry.getKey();
409                            if (key != null) {
410                                String propertyName = key.toString();
411                                if (propertyName.length() > 0) {
412                                    if (debug) log.debug(
413                                        "writeAMF3Object() - writing dynamic property: %s=%s",
414                                        propertyName, entry.getValue()
415                                    );
416                                    writeAMF3StringData(propertyName);
417                                    writeObject(entry.getValue());
418                                }
419                            }
420                        }
421                        writeAMF3StringData("");
422                    }
423                }
424            }
425    
426            if (debug) log.debug("writeAMF3Object(o=%s) - Done", o);
427        }

432        protected void addToStoredStrings(String s) {
433            if (!storedStrings.containsKey(s)) {
434                Integer index = Integer.valueOf(storedStrings.size());
435                if (debug) log.debug("addToStoredStrings(s=%s) at index=%d", s, index);
436                storedStrings.put(s, index);
437            }
438        }
439    
440        protected int indexOfStoredStrings(String s) {
441            Integer index = storedStrings.get(s);
442            if (debug) log.debug("indexOfStoredStrings(s=%s) -> %d", s, (index != null ? index : -1));
443            return (index != null ? index : -1);
444        }
445    
446        protected void addToStoredObjects(Object o) {
447            if (o != null && !storedObjects.containsKey(o)) {
448                Integer index = Integer.valueOf(storedObjects.size());
449                if (debug) log.debug("addToStoredObjects(o=%s) at index=%d", o, index);
450                storedObjects.put(o, index);
451            }
452        }
453    
454        protected int indexOfStoredObjects(Object o) {
455            Integer index = storedObjects.get(o);
456            if (debug) log.debug("indexOfStoredObjects(o=%s) -> %d", o, (index != null ? index : -1));
457            return (index != null ? index : -1);
458        }

460        protected IndexedJavaClassDescriptor addToStoredClassDescriptors(Class<?> clazz) {
461            final String name = JavaClassDescriptor.getClassName(clazz);
462    
463            if (debug) log.debug("addToStoredClassDescriptors(clazz=%s)", clazz);
464    
465            if (storedClassDescriptors.containsKey(name))
466                throw new RuntimeException(
467                    "Descriptor of \"" + name + "\" is already stored at index: " +
468                    getFromStoredClassDescriptors(clazz).getIndex()
469                );
470    
471            // find custom class descriptor and instantiate if any
472            JavaClassDescriptor desc = null;
473    
474            Class<? extends JavaClassDescriptor> descriptorType
475                = context.getGraniteConfig().getJavaDescriptor(clazz.getName());
476            if (descriptorType != null) {
477                Class<?>[] argsDef = new Class[]{Class.class};
478                Object[] argsVal = new Object[]{clazz};
479                try {
480                    desc = ClassUtil.newInstance(descriptorType, argsDef, argsVal);
481                } catch (Exception e) {
482                    throw new RuntimeException("Could not instantiate Java descriptor: " + descriptorType);
483                }
484            }
485    
486            if (desc == null)
487                desc = new DefaultJavaClassDescriptor(clazz);
488    
489            IndexedJavaClassDescriptor iDesc = new IndexedJavaClassDescriptor(storedClassDescriptors.size(), desc);
490    
491            if (debug) log.debug("addToStoredClassDescriptors() - putting: name=%s, iDesc=%s", name, iDesc);
492    
493            storedClassDescriptors.put(name, iDesc);
494    
495            return iDesc;
496        }
497    
498        protected IndexedJavaClassDescriptor getFromStoredClassDescriptors(Class<?> clazz) {
499            if (debug) log.debug("getFromStoredClassDescriptors(clazz=%s)", clazz);
500    
501            String name = JavaClassDescriptor.getClassName(clazz);
502            IndexedJavaClassDescriptor iDesc = storedClassDescriptors.get(name);
503    
504            if (debug) log.debug("getFromStoredClassDescriptors() -> %s", iDesc);
505    
506            return iDesc;
507        }

512        protected Channel getChannel() {
513            if (channel == null) {
514                String channelId = context.getAMFContext().getChannelId();
515                if (channelId != null) {
516                    channel = context.getServicesConfig().findChannelById(channelId);
517                    if (channel == null)
518                        log.warn("Could not get channel for channel id: %s", channelId);
519                }
520                else if (warnOnChannelIdMissing)
521                    log.warn("Could not get channel id for message: %s", context.getAMFContext().getRequest());
522            }
523            return channel;
524        }
