021    package org.granite.messaging.service.security;
022    
023    import org.granite.messaging.service.ServiceException;
024    
025    /**
026     * @author Franck WOLFF
027     */
028    public class SecurityServiceException extends ServiceException {
029    
030        private static final long serialVersionUID = 1L;
031    
032        /** Code for invalid credentails (wrong username or password) */
033        public static String CODE_INVALID_CREDENTIALS = "Server.Security.InvalidCredentials";
034        /** Code for illegal access to a service or method that requires authentication */
035        public static String CODE_NOT_LOGGED_IN = "Server.Security.NotLoggedIn";
036        /** Code for user session timeout */
037        public static String CODE_SESSION_EXPIRED = "Server.Security.SessionExpired";
038        /** Code for illegal access to a service or method that requires special role or profile */
039        public static String CODE_ACCESS_DENIED = "Server.Security.AccessDenied";
040    
041    
042        public static SecurityServiceException newInvalidCredentialsException() {
043            return new SecurityServiceException(CODE_INVALID_CREDENTIALS);
044        }
045        public static SecurityServiceException newInvalidCredentialsException(String message) {
046            return new SecurityServiceException(CODE_INVALID_CREDENTIALS, message);
047        }
048        public static SecurityServiceException newInvalidCredentialsException(String message, String details) {
049            return new SecurityServiceException(CODE_INVALID_CREDENTIALS, message, details);
050        }
051    
052        public static SecurityServiceException newNotLoggedInException() {
053            return new SecurityServiceException(CODE_NOT_LOGGED_IN);
054        }
055        public static SecurityServiceException newNotLoggedInException(String message) {
056            return new SecurityServiceException(CODE_NOT_LOGGED_IN, message);
057        }
058        public static SecurityServiceException newNotLoggedInException(String message, String details) {
059            return new SecurityServiceException(CODE_NOT_LOGGED_IN, message, details);
060        }
061    
062        public static SecurityServiceException newSessionExpiredException() {
063            return new SecurityServiceException(CODE_SESSION_EXPIRED);
064        }
065        public static SecurityServiceException newSessionExpiredException(String message) {
066            return new SecurityServiceException(CODE_SESSION_EXPIRED, message);
067        }
068        public static SecurityServiceException newSessionExpiredException(String message, String details) {
069            return new SecurityServiceException(CODE_SESSION_EXPIRED, message, details);
070        }
071    
072        public static SecurityServiceException newAccessDeniedException() {
073            return new SecurityServiceException(CODE_ACCESS_DENIED);
074        }
075        public static SecurityServiceException newAccessDeniedException(String message) {
076            return new SecurityServiceException(CODE_ACCESS_DENIED, message);
077        }
078        public static SecurityServiceException newAccessDeniedException(String message, String details) {
079            return new SecurityServiceException(CODE_ACCESS_DENIED, message, details);
080        }
081    
082    
083        public SecurityServiceException(String code) {
084            this(code, null, null);
085        }
086        public SecurityServiceException(String code, String message) {
087            this(code, message, null);
088        }
089        public SecurityServiceException(String code, String message, String details) {
090            super(code, message, details);
091        }
092    }
