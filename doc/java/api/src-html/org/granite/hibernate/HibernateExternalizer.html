084        static enum SerializeMetadata {
085            YES,
086            NO,
087            LAZY
088        }
089        
090        private SerializeMetadata serializeMetadata = SerializeMetadata.NO;

093        /**
094         * Configure this externalizer with the values supplied in granite-config.xml.
095         * 
096         * <p>The only supported configuration option is 'hibernate-collection-metadata' with
097         * values in ['no' (default), 'yes' and 'lazy']. By default, collection metadata (key,
098         * role and snapshot) aren't serialized. If the value of the 'hibernate-collection-metadata'
099         * node is 'yes', metadata will be always serialized, while the 'lazy' value tells the
100         * externalizer to serialiaze metadata for uninitialized collections only.
101         * 
102         * <p>Configuration example (granite-config.xml):
103         * <pre>
104         * &lt;granite-config scan="true"&gt;
105         *   &lt;externalizers&gt;
106         *     &lt;configuration&gt;
107         *       &lt;hibernate-collection-metadata&gt;lazy&lt;/hibernate-collection-metadata&gt;
108         *     &lt;/configuration&gt;
109         *   &lt;/externalizers&gt;
110         * &lt;/granite-config&gt;
111         * </pre>
112         * 
113         * @param properties an XMap instance that contains the configuration node.
114         */
115        @Override
116            public void configure(XMap properties) {
117            super.configure(properties);
118            
119            if (properties != null) {
120                    String collectionmetadata = properties.get("hibernate-collection-metadata");
121                    if (collectionmetadata != null) {
122                            if ("no".equalsIgnoreCase(collectionmetadata))
123                                    serializeMetadata = SerializeMetadata.NO;
124                            else if ("yes".equalsIgnoreCase(collectionmetadata))
125                                    serializeMetadata = SerializeMetadata.YES;
126                            else if ("lazy".equalsIgnoreCase(collectionmetadata))
127                                    serializeMetadata = SerializeMetadata.LAZY;
128                            else
129                                    throw new RuntimeException("Illegal value for the 'hibernate-collection-metadata' option: " + collectionmetadata);
130                    }
131            }
132            }

134            @Override
135        public Object newInstance(String type, ObjectInput in)
136            throws IOException, ClassNotFoundException, InstantiationException, InvocationTargetException, IllegalAccessException {
137    
138            // If type is not an entity (@Embeddable for example), we don't read initialized/detachedState
139            // and we fall back to DefaultExternalizer behavior.
140            Class<?> clazz = ClassUtil.forName(type);
141            if (!isRegularEntity(clazz))
142                return super.newInstance(type, in);
143            
144            // Read initialized flag.
145            boolean initialized = ((Boolean)in.readObject()).booleanValue();
146    
147            // Read detachedState.
148            String detachedState = (String)in.readObject();
149            
150            // New or initialized entity.
151            if (initialized)
152                return super.newInstance(type, in);
153    
154            // Actual proxy instantiation is deferred in order to keep consistent order in
155            // stored objects list (see AMF3Deserializer).
156            return newProxyInstantiator(proxyFactories, detachedState);
157        }
158            
159            protected Object newProxyInstantiator(ConcurrentHashMap<String, ProxyFactory> proxyFactories, String detachedState) {
160            return new HibernateProxyInstantiator(proxyFactories, detachedState);
161            }

163        @Override
164        public void readExternal(Object o, ObjectInput in) throws IOException, ClassNotFoundException, IllegalAccessException {
165    
166            // Skip unserialized fields for proxies (only read id).
167            if (o instanceof HibernateProxyInstantiator) {
168                    log.debug("Reading Hibernate Proxy...");
169                ((HibernateProxyInstantiator)o).readId(in);
170            }
171            // @Embeddable or others...
172            else if (!isRegularEntity(o.getClass())) {
173                    log.debug("Delegating non regular entity reading to DefaultExternalizer...");
174                super.readExternal(o, in);
175            }
176            // Regular @Entity or @MappedSuperclass
177            else {
178                GraniteConfig config = GraniteContext.getCurrentInstance().getGraniteConfig();
179    
180                Converters converters = config.getConverters();
181                ClassGetter classGetter = config.getClassGetter();
182                Class<?> oClass = classGetter.getClass(o);
183    
184                List<Property> fields = findOrderedFields(oClass, false);
185                log.debug("Reading entity %s with fields %s", oClass.getName(), fields);
186                for (Property field : fields) {
187                    Object value = in.readObject();
188                    if (!(field instanceof MethodProperty && field.isAnnotationPresent(ExternalizedProperty.class))) {
189                        
190                            if (value instanceof AbstractExternalizablePersistentCollection)
191                                    value = newHibernateCollection((AbstractExternalizablePersistentCollection)value, field);
192                        else if (!(value instanceof HibernateProxy))
193                            value = converters.convert(value, field.getType());
194    
195                            field.setProperty(o, value, false);
196                    }
197                }
198            }
199        }

201            protected PersistentCollection newHibernateCollection(AbstractExternalizablePersistentCollection value, Property field) {
202            final Type target = field.getType();
203            final boolean initialized = value.isInitialized();
204            final String metadata = value.getMetadata();
205                    final boolean dirty = value.isDirty();
206            final boolean sorted = (
207                    SortedSet.class.isAssignableFrom(ClassUtil.classOfType(target)) ||
208                    SortedMap.class.isAssignableFrom(ClassUtil.classOfType(target))
209            );
210            
211            Comparator<?> comparator = null;
212            if (sorted && field.isAnnotationPresent(Sort.class)) {
213                    Sort sort = field.getAnnotation(Sort.class);
214                    if (sort.type() == SortType.COMPARATOR) {
215                            try {
216                                    comparator = ClassUtil.newInstance(sort.comparator(), Comparator.class);
217                            } catch (Exception e) {
218                                    throw new RuntimeException("Could not create instance of Comparator: " + sort.comparator());
219                            }
220                    }
221            }
222            
223            PersistentCollection coll = null;
224                    if (value instanceof ExternalizablePersistentSet) {
225                    if (initialized) {
226                            Set<?> set = ((ExternalizablePersistentSet)value).getContentAsSet(target, comparator);
227                            coll = (sorted ? new PersistentSortedSet(null, (SortedSet<?>)set) : new PersistentSet(null, set));
228                }
229                    else
230                    coll = (sorted ? new PersistentSortedSet() : new PersistentSet());
231            }
232                    else if (value instanceof ExternalizablePersistentBag) {
233                    if (initialized) {
234                        List<?> bag = ((ExternalizablePersistentBag)value).getContentAsList(target);
235                    coll = new PersistentBag(null, bag);
236                    }
237                    else
238                        coll = new PersistentBag();
239                    }
240                    else if (value instanceof ExternalizablePersistentList) {
241                    if (initialized) {
242                        List<?> list = ((ExternalizablePersistentList)value).getContentAsList(target);
243                    coll = new PersistentList(null, list);
244                    }
245                    else
246                        coll = new PersistentList();
247                    }
248                    else if (value instanceof ExternalizablePersistentMap) {
249                    if (initialized) {
250                        Map<?, ?> map = ((ExternalizablePersistentMap)value).getContentAsMap(target, comparator);
251                        coll = (sorted ? new PersistentSortedMap(null, (SortedMap<?, ?>)map) : new PersistentMap(null, map));
252                    }
253                    else
254                        coll = (sorted ? new PersistentSortedMap() : new PersistentMap());
255                    }
256                    else
257                            throw new RuntimeException("Illegal externalizable persitent class: " + value);
258                    
259                    if (metadata != null && serializeMetadata != SerializeMetadata.NO && (serializeMetadata == SerializeMetadata.YES || !initialized)) {
260                    String[] toks = metadata.split(":", 3);
261                    if (toks.length != 3)
262                            throw new RuntimeException("Invalid collection metadata: " + metadata);
263                    Serializable key = deserializeSerializable(StringUtil.hexStringToBytes(toks[0]));
264                    Serializable snapshot = deserializeSerializable(StringUtil.hexStringToBytes(toks[1]));
265                    String role = toks[2];
266                coll.setSnapshot(key, role, snapshot);
267                    }
268                    
269                    if (initialized && dirty)
270                            coll.dirty();
271            
272            return coll;
273        }

275        @Override
276        public void writeExternal(Object o, ObjectOutput out) throws IOException, IllegalAccessException {
277    
278            ClassGetter classGetter = GraniteContext.getCurrentInstance().getGraniteConfig().getClassGetter();
279            Class<?> oClass = classGetter.getClass(o);
280    
281            if (o instanceof HibernateProxy) {              
282                HibernateProxy proxy = (HibernateProxy)o;
283    
284                // Only write initialized flag & detachedState & entity id if proxy is uninitialized.
285                if (proxy.getHibernateLazyInitializer().isUninitialized()) {
286                    String detachedState = getProxyDetachedState(proxy);
287                    Serializable id = proxy.getHibernateLazyInitializer().getIdentifier();
288                    log.debug("Writing uninitialized HibernateProxy %s with id %s", detachedState, id);
289                    
290                    // Write initialized flag.
291                    out.writeObject(Boolean.FALSE);
292                    // Write detachedState.
293                    out.writeObject(detachedState);
294                    // Write entity id.
295                    out.writeObject(id);
296                    return;
297                }
298    
299                // Proxy is initialized, get the underlying persistent object.
300                    log.debug("Writing initialized HibernateProxy...");
301                o = proxy.getHibernateLazyInitializer().getImplementation();
302            }
303    
304            if (!isRegularEntity(o.getClass())) { // @Embeddable or others...
305                    log.debug("Delegating non regular entity writing to DefaultExternalizer...");
306                super.writeExternal(o, out);
307            }
308            else {
309                // Write initialized flag.
310                out.writeObject(Boolean.TRUE);
311                // Write detachedState.
312                out.writeObject(null);
313    
314                // Externalize entity fields.
315                List<Property> fields = findOrderedFields(oClass, false);
316                log.debug("Writing entity %s with fields %s", o.getClass().getName(), fields);
317                for (Property field : fields) {
318                    Object value = field.getProperty(o);
319                    if (value instanceof PersistentCollection)
320                            value = newExternalizableCollection((PersistentCollection)value);
321                    if (isValueIgnored(value))
322                            out.writeObject(null);
323                    else
324                            out.writeObject(value);
325                }
326            }
327        }

329        protected AbstractExternalizablePersistentCollection newExternalizableCollection(PersistentCollection value) {
330            final boolean initialized = Hibernate.isInitialized(value);
331            final boolean dirty = value.isDirty();
332            
333            AbstractExternalizablePersistentCollection coll = null;
334            
335            if (value instanceof PersistentSet)
336                coll = new ExternalizablePersistentSet(initialized ? (Set<?>)value : null, initialized, dirty);
337            else if (value instanceof PersistentList)
338                coll = new ExternalizablePersistentList(initialized ? (List<?>)value : null, initialized, dirty);
339            else if (value instanceof PersistentBag)
340                coll = new ExternalizablePersistentBag(initialized ? (List<?>)value : null, initialized, dirty);
341            else if (value instanceof PersistentMap)
342                coll = new ExternalizablePersistentMap(initialized ? (Map<?, ?>)value : null, initialized, dirty);
343            else
344                throw new UnsupportedOperationException("Unsupported Hibernate collection type: " + value);
345    
346            if (serializeMetadata != SerializeMetadata.NO && (serializeMetadata == SerializeMetadata.YES || !initialized) && value.getRole() != null) {
347                    char[] hexKey = StringUtil.bytesToHexChars(serializeSerializable(value.getKey()));
348                    char[] hexSnapshot = StringUtil.bytesToHexChars(serializeSerializable(value.getStoredSnapshot()));
349                    String metadata = new StringBuilder(hexKey.length + 1 + hexSnapshot.length + 1 + value.getRole().length())
350                                    .append(hexKey).append(':')
351                            .append(hexSnapshot).append(':')
352                            .append(value.getRole())
353                            .toString();
354                    coll.setMetadata(metadata);
355            }
356            
357            return coll;
358        }

360        @Override
361        public int accept(Class<?> clazz) {
362            return (
363                clazz.isAnnotationPresent(Entity.class) ||
364                clazz.isAnnotationPresent(MappedSuperclass.class) ||
365                clazz.isAnnotationPresent(Embeddable.class)
366            ) ? 1 : -1;
367        }
368    
369        protected String getProxyDetachedState(HibernateProxy proxy) {
370            LazyInitializer initializer = proxy.getHibernateLazyInitializer();
371    
372            StringBuilder sb = new StringBuilder();
373    
374            sb.append(initializer.getClass().getName())
375              .append(':');
376            if (initializer.getPersistentClass() != null)
377                sb.append(initializer.getPersistentClass().getName());
378            sb.append(':');
379            if (initializer.getEntityName() != null)
380                sb.append(initializer.getEntityName());
381    
382            return sb.toString();
383        }
384    
385        protected boolean isRegularEntity(Class<?> clazz) {
386            return clazz.isAnnotationPresent(Entity.class) || clazz.isAnnotationPresent(MappedSuperclass.class);
387        }
388        
389        protected byte[] serializeSerializable(Serializable o) {
390            if (o == null)
391                    return BYTES_0;
392            try {
393                    ByteArrayOutputStream baos = new ByteArrayOutputStream();
394                    ObjectOutputStream oos = new ObjectOutputStream(baos);
395                    oos.writeObject(o);
396                    return baos.toByteArray();
397            } catch (Exception e) {
398                    throw new RuntimeException("Could not serialize: " + o);
399            }
400        }
401        
402        protected Serializable deserializeSerializable(byte[] data) {
403            if (data.length == 0)
404                    return null;
405            try {
406                    ByteArrayInputStream baos = new ByteArrayInputStream(data);
407                    ObjectInputStream oos = new ObjectInputStream(baos);
408                    return (Serializable)oos.readObject();
409            } catch (Exception e) {
410                    throw new RuntimeException("Could not deserialize: " + data);
411            }
412        }
