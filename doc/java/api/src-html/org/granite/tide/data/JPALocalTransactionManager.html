021    package org.granite.tide.data;
022    
023    import javax.persistence.EntityTransaction;
024    
025    import org.granite.tide.TideTransactionManager;
026    
027    
028    /**
029     * Responsible for attaching a entity with the entity mangager
030     * @author cingram
031     *
032     */
033    public class JPALocalTransactionManager implements TideTransactionManager {
034            
035            public Object begin(TideTransactionPersistenceManager pm) {
036                    if (pm != null)
037                            return pm.getCurrentTransaction();
038                    return null;
039            }
040            
041            public void commit(Object tx) throws Exception {
042            ((EntityTransaction)tx).commit();
043            }
044            
045            public void rollback(Object tx) {
046            ((EntityTransaction)tx).rollback();
047            }
048    }
