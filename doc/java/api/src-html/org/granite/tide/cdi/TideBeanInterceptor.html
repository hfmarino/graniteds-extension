021    package org.granite.tide.cdi;
022    
023    import javax.inject.Inject;
024    import javax.inject.Named;
025    import javax.interceptor.AroundInvoke;
026    import javax.interceptor.Interceptor;
027    import javax.interceptor.InvocationContext;
028    
029    import org.granite.messaging.amf.io.util.externalizer.DefaultExternalizer;
030    import org.granite.messaging.amf.io.util.externalizer.annotation.ExternalizedBean;
031    
032    
033    @TideBean @ExternalizedBean(type=DefaultExternalizer.class) @Interceptor
034    public class TideBeanInterceptor {
035    
036            @Inject
037            CDIServiceContext tideContext;
038            
039            @AroundInvoke
040            public Object doAround(InvocationContext context) throws Exception {
041                    Object result = context.proceed();
042                    
043                    String componentName = null;
044                    if (context.getTarget().getClass().isAnnotationPresent(Named.class))
045                            componentName = context.getTarget().getClass().getAnnotation(Named.class).value();
046                    
047            tideContext.addResultEval(new ScopedContextResult(componentName, null, context.getTarget()));
048                    
049                    return result;
050            }
051    }
