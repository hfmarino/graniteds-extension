021    package org.granite.tide;
022    
023    import java.util.Collections;
024    import java.util.List;
025    import java.util.Map;
026    
027    /**
028     * @author William DRAI
029     */
030    public class TideStatusMessages {
031        
032            private static final List<TideMessage> EMPTY_MESSAGES = Collections.emptyList();
033            private static final Map<String, List<TideMessage>> EMPTY_KEYED_MESSAGES = Collections.emptyMap();
034        public static final TideStatusMessages EMPTY_STATUS_MESSAGES = new TideStatusMessages(EMPTY_MESSAGES, EMPTY_KEYED_MESSAGES);
035        
036            private List<TideMessage> messages;
037            private Map<String, List<TideMessage>> keyedMessages;
038            
039        
040        public TideStatusMessages(List<TideMessage> messages) {
041            this.messages = messages;
042            this.keyedMessages = EMPTY_KEYED_MESSAGES;
043        }
044        
045        public TideStatusMessages(List<TideMessage> messages, Map<String, List<TideMessage>> keyedMessages) {
046            this.messages = messages;
047            this.keyedMessages = keyedMessages;
048        }
049        
050        public List<TideMessage> getMessages() {
051            return messages;
052        }
053        
054        public Map<String, List<TideMessage>> getKeyedMessages() {
055            return keyedMessages;
056        }
057    }
