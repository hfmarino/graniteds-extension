021    package org.granite.gravity;
022    
023    
024    /**
025     * @author Franck WOLFF
026     */
027    public class MessagePublishingException extends Exception {
028    
029            private static final long serialVersionUID = 1L;
030            
031            private final AsyncPublishedMessage unpublishedMessage;
032    
033            public MessagePublishingException(AsyncPublishedMessage unpublishedMessage) {
034                    this.unpublishedMessage = unpublishedMessage;
035            }
036    
037            public MessagePublishingException(AsyncPublishedMessage unpublishedMessage, String message) {
038                    super(message);
039                    this.unpublishedMessage = unpublishedMessage;
040            }
041    
042            public MessagePublishingException(AsyncPublishedMessage unpublishedMessage, Throwable cause) {
043                    super(cause);
044                    this.unpublishedMessage = unpublishedMessage;
045            }
046    
047            public MessagePublishingException(AsyncPublishedMessage unpublishedMessage, String message, Throwable cause) {
048                    super(message, cause);
049                    this.unpublishedMessage = unpublishedMessage;
050            }
051    
052            public AsyncPublishedMessage getUnreceivedMessage() {
053                    return unpublishedMessage;
054            }
055    }
