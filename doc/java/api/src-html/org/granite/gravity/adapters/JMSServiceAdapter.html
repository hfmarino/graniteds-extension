062        private static final Logger log = Logger.getLogger(JMSServiceAdapter.class);

064        protected ConnectionFactory jmsConnectionFactory = null;
065        protected javax.jms.Destination jmsDestination = null;
066        protected Map<String, JMSClient> jmsClients = new HashMap<String, JMSClient>();
067        protected String destinationName = null;
068        protected boolean textMessages = false;
069        protected boolean transactedSessions = false;
070        protected int acknowledgeMode = Session.AUTO_ACKNOWLEDGE;
071        protected int messagePriority = javax.jms.Message.DEFAULT_PRIORITY;
072        protected int deliveryMode = javax.jms.Message.DEFAULT_DELIVERY_MODE;
073        protected boolean noLocal = false;

075        @Override
076        public void configure(XMap adapterProperties, XMap destinationProperties) throws ServiceException {
077            super.configure(adapterProperties, destinationProperties);
078    
079            try {
080                destinationName = destinationProperties.get("jms/destination-name");
081                if (Boolean.TRUE.toString().equals(destinationProperties.get("jms/transacted-sessions")))
082                    transactedSessions = true;
083                if ("AUTO_ACKNOWLEDGE".equals(destinationProperties.get("jms/acknowledge-mode")))
084                    acknowledgeMode = Session.AUTO_ACKNOWLEDGE;
085                else if ("CLIENT_ACKNOWLEDGE".equals(destinationProperties.get("jms/acknowledge-mode")))
086                    acknowledgeMode = Session.CLIENT_ACKNOWLEDGE;
087                else if ("DUPS_OK_ACKNOWLEDGE".equals(destinationProperties.get("jms/acknowledge-mode")))
088                    acknowledgeMode = Session.DUPS_OK_ACKNOWLEDGE;
089                if ("javax.jms.TextMessage".equals(destinationProperties.get("jms/message-type")))
090                    textMessages = true;
091                if (Boolean.TRUE.toString().equals(destinationProperties.get("jms/no-local")))
092                    noLocal = true;
093    
094                Properties environment = new Properties();
095                for (XMap property : destinationProperties.getAll("jms/initial-context-environment/property")) {
096                    String name = property.get("name");
097                    String value = property.get("value");
098                    
099                    if ("Context.PROVIDER_URL".equals(name))
100                            environment.put(Context.PROVIDER_URL, value);
101                    else if ("Context.INITIAL_CONTEXT_FACTORY".equals(name))
102                            environment.put(Context.INITIAL_CONTEXT_FACTORY, value);
103                    else if ("Context.URL_PKG_PREFIXES".equals(name))
104                            environment.put(Context.URL_PKG_PREFIXES, value);
105                    else if ("Context.SECURITY_PRINCIPAL".equals(name))
106                            environment.put(Context.SECURITY_PRINCIPAL, value);
107                    else if ("Context.SECURITY_CREDENTIALS".equals(name))
108                            environment.put(Context.SECURITY_CREDENTIALS, value);
109                    else
110                            log.warn("Unknown InitialContext property: %s (ignored)", name);
111                }
112    
113                InitialContext ic = new InitialContext(environment.size() > 0 ? environment : null);
114                
115                String cfJndiName = destinationProperties.get("jms/connection-factory");
116                jmsConnectionFactory = (ConnectionFactory)ic.lookup(cfJndiName);
117    
118                String dsJndiName = destinationProperties.get("jms/destination-jndi-name");
119                jmsDestination = (Destination)ic.lookup(dsJndiName);
120            }
121            catch (Exception e) {
122                throw new ServiceException("Error when configuring JMS Adapter", e);
123            }
124        }

127            return jmsDestination;
128        }
129

131            return jmsDestination;

148        private synchronized JMSClient createJMSClient(Channel client) throws Exception {
149            JMSClient jmsClient = jmsClients.get(client.getId());
150            if (jmsClient == null) {
151                jmsClient = new JMSClient(client);
152                jmsClient.connect();
153                jmsClients.put(client.getId(), jmsClient);
154            }
155            return jmsClient;
156        }
157    
158        @Override
159        public Object invoke(Channel fromClient, AsyncMessage message) {
160            try {
161                JMSClient jmsClient = createJMSClient(fromClient);
162                jmsClient.send(message);
163    
164                AsyncMessage reply = new AcknowledgeMessage(message);
165                reply.setMessageId(message.getMessageId());
166    
167                return reply;
168            }
169            catch (Exception e) {
170                    log.error(e, "Error sending message");
171                ErrorMessage error = new ErrorMessage(message, null);
172                error.setFaultString("JMS Adapter error " + e.getMessage());
173    
174                return error;
175            }
176        }
177    
178        @Override
179        public Object manage(Channel fromChannel, CommandMessage message) {
180            if (message.getOperation() == CommandMessage.SUBSCRIBE_OPERATION) {
181                try {
182                    JMSClient jmsClient = createJMSClient(fromChannel);
183                    jmsClient.subscribe(message);
184                    
185                    AsyncMessage reply = new AcknowledgeMessage(message);
186                    return reply;
187                }
188                catch (Exception e) {
189                    throw new RuntimeException("JMSAdapter subscribe error on topic: " + message, e);
190                }
191            }
192            else if (message.getOperation() == CommandMessage.UNSUBSCRIBE_OPERATION) {
193                try {
194                    JMSClient jmsClient = createJMSClient(fromChannel);
195                    jmsClient.unsubscribe(message);
196    
197                    AsyncMessage reply = new AcknowledgeMessage(message);
198                    return reply;
199                }
200                catch (Exception e) {
201                    throw new RuntimeException("JMSAdapter unsubscribe error on topic: " + message, e);
202                }
203            }
204    
205            return null;
206        }

209        private class JMSClient {
210    
211            private Channel channel = null;
212            private String topic = null;
213            private javax.jms.Connection jmsConnection = null;
214            private javax.jms.Session jmsProducerSession = null;
215            private javax.jms.MessageProducer jmsProducer = null;
216            private Map<String, JMSConsumer> consumers = new HashMap<String, JMSConsumer>();
217    
218    
219            public JMSClient(Channel channel) {
220                this.channel = channel;
221            }
222    
223            public void connect() throws ServiceException {
224                try {
225                    jmsConnection = jmsConnectionFactory.createConnection();
226                    jmsConnection.start();
227                }
228                catch (JMSException e) {
229                    throw new ServiceException("JMS Initialize error", e);
230                }
231            }
232    
233            public void close() throws ServiceException {
234                try {
235                    if (jmsProducer != null)
236                        jmsProducer.close();
237                    if (jmsProducerSession != null)
238                        jmsProducerSession.close();
239                    for (JMSConsumer consumer : consumers.values())
240                        consumer.close();
241                    jmsConnection.stop();
242                    jmsConnection.close();
243                }
244                catch (JMSException e) {
245                    throw new ServiceException("JMS Stop error", e);
246                }
247            }
248            
249    
250            public void send(Message message) throws Exception {
251                if (jmsProducerSession == null)
252                    jmsProducerSession = jmsConnection.createSession(transactedSessions, acknowledgeMode);
253                if (jmsProducer == null) {
254                    jmsProducer = jmsProducerSession.createProducer(getProducerDestination(topic));
255                    jmsProducer.setPriority(messagePriority);
256                    jmsProducer.setDeliveryMode(deliveryMode);
257                }
258                
259                Object msg = null;
260                if (Boolean.TRUE.equals(message.getHeader(Gravity.BYTEARRAY_BODY_HEADER))) {
261                    byte[] byteArray = (byte[])message.getBody();
262                    ByteArrayInputStream bais = new ByteArrayInputStream(byteArray);
263                    AMF3Deserializer deser = new AMF3Deserializer(bais);
264                    msg = deser.readObject();
265                }
266                else
267                    msg = message.getBody();
268    
269                javax.jms.Message jmsMessage = null;
270                if (textMessages)
271                    jmsMessage = jmsProducerSession.createTextMessage(msg.toString());
272                else
273                    jmsMessage = jmsProducerSession.createObjectMessage((Serializable)msg);
274    
275                jmsMessage.setJMSMessageID(normalizeJMSMessageID(message.getMessageId()));
276                jmsMessage.setJMSCorrelationID(normalizeJMSMessageID(((AsyncMessage)message).getCorrelationId()));
277                jmsMessage.setJMSTimestamp(message.getTimestamp());
278                jmsMessage.setJMSExpiration(message.getTimeToLive());
279    
280                for (Map.Entry<String, Object> me : message.getHeaders().entrySet()) {
281                    if ("JMSType".equals(me.getKey())) {
282                        if (me.getValue() instanceof String)
283                            jmsMessage.setJMSType((String)me.getValue());
284                    }
285                    else if ("JMSPriority".equals(me.getKey())) {
286                        if (me.getValue() instanceof Integer)
287                            jmsMessage.setJMSPriority(((Integer)me.getValue()).intValue());
288                    }
289                    else if (me.getValue() instanceof String)
290                        jmsMessage.setStringProperty(me.getKey(), (String)me.getValue());
291                    else if (me.getValue() instanceof Boolean)
292                        jmsMessage.setBooleanProperty(me.getKey(), ((Boolean)me.getValue()).booleanValue());
293                    else if (me.getValue() instanceof Integer)
294                        jmsMessage.setIntProperty(me.getKey(), ((Integer)me.getValue()).intValue());
295                    else if (me.getValue() instanceof Long)
296                        jmsMessage.setLongProperty(me.getKey(), ((Long)me.getValue()).longValue());
297                    else if (me.getValue() instanceof Double)
298                        jmsMessage.setDoubleProperty(me.getKey(), ((Double)me.getValue()).doubleValue());
299                    else
300                        jmsMessage.setObjectProperty(me.getKey(), me.getValue());
301                }
302    
303                jmsProducer.send(jmsMessage);
304                if (transactedSessions)
305                    jmsProducerSession.commit();
306            }
307    
308                    private String normalizeJMSMessageID(String messageId) {
309                if (messageId != null && !messageId.startsWith("ID:"))
310                    messageId = "ID:" + messageId;
311                            return messageId;
312                    }
313    
314            public void subscribe(Message message) throws Exception {
315                String subscriptionId = (String)message.getHeader(AsyncMessage.DESTINATION_CLIENT_ID_HEADER);
316                String selector = (String)message.getHeader(CommandMessage.SELECTOR_HEADER);
317                this.topic = (String)message.getHeader(AsyncMessage.SUBTOPIC_HEADER);
318    
319                synchronized (consumers) {
320                    JMSConsumer consumer = consumers.get(subscriptionId);
321                    if (consumer == null) {
322                        consumer = new JMSConsumer(subscriptionId, selector, noLocal);
323                        consumers.put(subscriptionId, consumer);
324                    }
325                    else
326                        consumer.setSelector(selector);
327                    channel.addSubscription(message.getDestination(), topic, subscriptionId, false);
328                }
329            }
330    
331            public void unsubscribe(Message message) throws Exception {
332                String subscriptionId = (String)message.getHeader(AsyncMessage.DESTINATION_CLIENT_ID_HEADER);
333    
334                synchronized (consumers) {
335                    JMSConsumer consumer = consumers.get(subscriptionId);
336                    if (consumer != null)
337                        consumer.close();
338                    consumers.remove(subscriptionId);
339                    channel.removeSubscription(subscriptionId);
340                }
341            }

344            private class JMSConsumer implements MessageListener {
345    
346                private String subscriptionId = null;
347                private javax.jms.Session jmsConsumerSession = null;
348                private javax.jms.MessageConsumer jmsConsumer = null;
349                private boolean noLocal = false;
350    
351                public JMSConsumer(String subscriptionId, String selector, boolean noLocal) throws JMSException {
352                    this.subscriptionId = subscriptionId;
353                    this.noLocal = noLocal;
354                    jmsConsumerSession = jmsConnection.createSession(transactedSessions, acknowledgeMode);
355                    jmsConsumer = jmsConsumerSession.createConsumer(getConsumerDestination(topic), selector, noLocal);
356                    jmsConsumer.setMessageListener(this);
357                }
358    
359                public void setSelector(String selector) throws JMSException {
360                    if (jmsConsumer != null)
361                        jmsConsumer.close();
362                    jmsConsumer = jmsConsumerSession.createConsumer(getConsumerDestination(topic), selector, noLocal);
363                    jmsConsumer.setMessageListener(this);
364                }
365    
366                public void close() throws JMSException {
367                    if (jmsConsumer != null)
368                        jmsConsumer.close();
369                    if (jmsConsumerSession != null)
370                        jmsConsumerSession.close();
371                }
372    
373                public void onMessage(javax.jms.Message message) {
374                    if (!(message instanceof ObjectMessage) && !(message instanceof TextMessage)) {
375                        log.error("JMS Adapter message type not allowed: %s", message.getClass().getName());
376    
377                        try {
378                            if (acknowledgeMode == Session.CLIENT_ACKNOWLEDGE)
379                                message.acknowledge();
380    
381                            if (transactedSessions)
382                                jmsConsumerSession.commit();
383                        }
384                        catch (JMSException e) {
385                            log.error(e, "Could not ack/commit JMS onMessage");
386                        }
387                    }
388    
389                    log.debug("Delivering JMS message");
390    
391                    AsyncMessage dmsg = new AsyncMessage();
392                    try {
393                        Serializable msg = null;
394    
395                        if (textMessages) {
396                            TextMessage jmsMessage = (TextMessage)message;
397                            msg = jmsMessage.getText();
398                        }
399                        else {
400                            ObjectMessage jmsMessage = (ObjectMessage)message;
401                            msg = jmsMessage.getObject();
402                        }
403    
404                        dmsg.setDestination(getDestination().getId());
405                        
406                        if (Boolean.TRUE.equals(message.getBooleanProperty(Gravity.BYTEARRAY_BODY_HEADER))) {
407                            getGravity().initThread();
408                            try {
409                                    ByteArrayOutputStream baos = new ByteArrayOutputStream(100);
410                                    AMF3Serializer ser = new AMF3Serializer(baos, false);
411                                    ser.writeObject(msg);
412                                    ser.close();
413                                    baos.close();
414                                    dmsg.setBody(baos.toByteArray());
415                            }
416                            finally {
417                                    getGravity().releaseThread();
418                            }
419                        }
420                        else
421                            dmsg.setBody(msg);
422                        
423                        dmsg.setMessageId(denormalizeJMSMessageID(message.getJMSMessageID()));
424                        dmsg.setCorrelationId(denormalizeJMSMessageID(message.getJMSCorrelationID()));
425                        dmsg.setTimestamp(message.getJMSTimestamp());
426                        dmsg.setTimeToLive(message.getJMSExpiration());
427    
428                        Enumeration<?> ename = message.getPropertyNames();
429                        while (ename.hasMoreElements()) {
430                            String pname = (String)ename.nextElement();
431                            dmsg.setHeader(pname, message.getObjectProperty(pname));
432                        }
433                            
434                        dmsg.setHeader("JMSType", message.getJMSType());
435                        dmsg.setHeader("JMSPriority", Integer.valueOf(message.getJMSPriority()));
436                        dmsg.setHeader("JMSRedelivered", Boolean.valueOf(message.getJMSRedelivered()));
437                        dmsg.setHeader("JMSDeliveryMode", Integer.valueOf(message.getJMSDeliveryMode()));
438                        dmsg.setHeader(AsyncMessage.DESTINATION_CLIENT_ID_HEADER, subscriptionId);
439                        
440                        channel.receive(dmsg);
441                    }
442                    catch (IOException e) {
443                        if (transactedSessions) {
444                            try {
445                                jmsConsumerSession.rollback();
446                            }
447                            catch (JMSException f) {
448                                log.error("Could not rollback JMS session, messageId: %s", dmsg.getMessageId());
449                            }
450                        }
451    
452                        throw new RuntimeException("IO Error", e);
453                    }
454                    catch (JMSException e) {
455                        if (transactedSessions) {
456                            try {
457                                jmsConsumerSession.rollback();
458                            }
459                            catch (JMSException f) {
460                                log.error("Could not rollback JMS session, messageId: %s", dmsg.getMessageId());
461                            }
462                        }
463    
464                        throw new RuntimeException("JMS Error", e);
465                    }
466                    catch (MessageReceivingException e) {
467                        if (transactedSessions) {
468                            try {
469                                jmsConsumerSession.rollback();
470                            }
471                            catch (JMSException f) {
472                                log.error("Could not rollback JMS session, messageId: %s", dmsg.getMessageId());
473                            }
474                        }
475    
476                        throw new RuntimeException("Channel delivery Error", e);
477                    }
478    
479                    try {
480                        if (acknowledgeMode == Session.CLIENT_ACKNOWLEDGE)
481                            message.acknowledge();
482    
483                        if (transactedSessions)
484                            jmsConsumerSession.commit();
485                    }
486                    catch (JMSException e) {
487                        log.error("Could not ack/commit JMS onMessage, messageId: %s", dmsg.getMessageId());
488    
489                        // Message already delivered, should rollback or not ?
490                    }
491                }
492    
493                    private String denormalizeJMSMessageID(String messageId) {
494                    if (messageId != null && messageId.startsWith("ID:"))
495                            messageId = messageId.substring(3);
496                            return messageId;
497                    }
498            }
