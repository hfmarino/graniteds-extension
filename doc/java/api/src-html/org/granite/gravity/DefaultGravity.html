120        public void start() throws Exception {
121            log.info("Starting Gravity...");
122            synchronized (this) {
123                    if (!started) {
124                        adapterFactory = new AdapterFactory(this);
125                        internalStart();
126                        serverChannel = new ServerChannel(this, gravityConfig, ServerChannel.class.getName());
127                        started = true;
128                    }
129            }
130            log.info("Gravity successfully started.");
131        }
132        
133        protected void internalStart() {
134            gravityPool = new GravityPool(gravityConfig);
135            channelsTimer = new Timer();
136            
137            if (graniteConfig.isRegisterMBeans()) {
138                    try {
139                        ObjectName name = new ObjectName("org.granite:type=Gravity,context=" + graniteConfig.getMBeanContextName());
140                            log.info("Registering MBean: %s", name);
141                        OpenMBean mBean = OpenMBean.createMBean(this);
142                            MBeanServerLocator.getInstance().register(mBean, name, true);
143                    }
144                    catch (Exception e) {
145                            log.error(e, "Could not register Gravity MBean for context: %s", graniteConfig.getMBeanContextName());
146                    }
147            }
148        }

168        public void stop(boolean now) throws Exception {
169            log.info("Starting Gravity (now=%s)...", now);
170            synchronized (this) {
171                    if (adapterFactory != null) {
172                        try {
173                                            adapterFactory.stopAll();
174                                    } catch (Exception e) {
175                                    log.error(e, "Error while stopping adapter factory");
176                                    }
177                        adapterFactory = null;
178                    }
179    
180                    if (serverChannel != null) {
181                        try {
182                                            removeChannel(serverChannel.getId());
183                                    } catch (Exception e) {
184                                    log.error(e, "Error while removing server channel: %s", serverChannel);
185                                    }
186                        serverChannel = null;
187                    }
188                
189                if (channelsTimer != null) {
190                        try {
191                                            channelsTimer.cancel();
192                                    } catch (Exception e) {
193                                    log.error(e, "Error while cancelling channels timer");
194                                    }
195                        channelsTimer = null;
196                }
197                    
198                    if (gravityPool != null) {
199                            try {
200                                    if (now)
201                                            gravityPool.shutdownNow();
202                                    else
203                                            gravityPool.shutdown();
204                            }
205                            catch (Exception e) {
206                                    log.error(e, "Error while stopping thread pool");
207                            }
208                            gravityPool = null;
209                    }
210                
211                started = false;
212            }
213            log.info("Gravity sucessfully stopped.");
214        }

320        protected Channel createChannel() {
321            Channel channel = gravityConfig.getChannelFactory().newChannel(UUIDUtil.randomUUID());
322            TimeChannel timeChannel = new TimeChannel(channel);
323            for (int i = 0; channels.putIfAbsent(channel.getId(), timeChannel) != null; i++) {
324                if (i >= 10)
325                    throw new RuntimeException("Could not find random new clientId after 10 iterations");
326                channel.destroy();
327                channel = gravityConfig.getChannelFactory().newChannel(UUIDUtil.randomUUID());
328                timeChannel = new TimeChannel(channel);
329            }
330            
331            // Initialize timer task.
332            access(channel.getId());
333            
334            return channel;
335        }
336    
337            public Channel getChannel(String channelId) {
338            if (channelId == null)
339                return null;
340    
341            TimeChannel timeChannel = channels.get(channelId);
342            if (timeChannel != null)
343                    return timeChannel.getChannel();
344    
345            return null;
346        }
347    
348        public Channel removeChannel(String channelId) {
349            if (channelId == null)
350                return null;
351    
352            TimeChannel timeChannel = channels.get(channelId);
353            Channel channel = null;
354            if (timeChannel != null) {
355                    try {
356                            if (timeChannel.getTimerTask() != null)
357                                    timeChannel.getTimerTask().cancel();
358                    }
359                    catch (Exception e) {
360                            // Should never happen...
361                    }
362                    
363                    channel = timeChannel.getChannel();
364                    
365                    try {
366                        for (Subscription subscription : channel.getSubscriptions()) {
367                            try {
368                                    Message message = subscription.getUnsubscribeMessage();
369                                    handleMessage(message, true);
370                            }
371                            catch (Exception e) {
372                                    log.error(e, "Error while unsubscribing channel: %s from subscription: %s", channel, subscription);
373                            }
374                        }
375                    }
376                    finally {
377                            try {
378                                    channel.destroy();
379                            }
380                            finally {
381                                    channels.remove(channelId);
382                            }
383                    }
384            }
385    
386            return channel;
387        }
388        
389        public boolean access(String channelId) {
390            if (channelId != null) {
391                    TimeChannel timeChannel = channels.get(channelId);
392                    if (timeChannel != null) {
393                            synchronized (timeChannel) {
394                                    TimerTask timerTask = timeChannel.getTimerTask();
395                                    if (timerTask != null) {
396                                        log.debug("Canceling TimerTask: %s", timerTask);
397                                        timerTask.cancel();
398                                            timeChannel.setTimerTask(null);
399                                    }
400                                
401                                    timerTask = new ChannelTimerTask(this, channelId);
402                                timeChannel.setTimerTask(timerTask);
403                                
404                                long timeout = gravityConfig.getChannelIdleTimeoutMillis();
405                                log.debug("Scheduling TimerTask: %s for %s ms.", timerTask, timeout);
406                                channelsTimer.schedule(timerTask, timeout);
407                                return true;
408                            }
409                    }
410            }
411            return false;
412        }

437        public Message handleMessage(Message message, boolean skipInterceptor) {
438    
439            AMF3MessageInterceptor interceptor = null;
440            if (!skipInterceptor)
441                    interceptor = GraniteContext.getCurrentInstance().getGraniteConfig().getAmf3MessageInterceptor();
442            
443            Message reply = null;
444            
445            try {
446                    if (interceptor != null)
447                        interceptor.before(message);
448    
449                    if (message instanceof CommandMessage) {
450                        CommandMessage command = (CommandMessage)message;
451            
452                        switch (command.getOperation()) {
453            
454                        case CommandMessage.LOGIN_OPERATION:
455                        case CommandMessage.LOGOUT_OPERATION:
456                            return handleSecurityMessage(command);
457            
458                        case CommandMessage.CLIENT_PING_OPERATION:
459                            return handlePingMessage(command);
460                        case CommandMessage.CONNECT_OPERATION:
461                            return handleConnectMessage(command);
462                        case CommandMessage.DISCONNECT_OPERATION:
463                            return handleDisconnectMessage(command);
464                        case CommandMessage.SUBSCRIBE_OPERATION:
465                            return handleSubscribeMessage(command);
466                        case CommandMessage.UNSUBSCRIBE_OPERATION:
467                            return handleUnsubscribeMessage(command);
468            
469                        default:
470                            throw new UnsupportedOperationException("Unsupported command operation: " + command);
471                        }
472                    }
473            
474                    reply = handlePublishMessage((AsyncMessage)message);
475            }
476            finally {
477                    if (interceptor != null)
478                        interceptor.after(message, reply);
479            }
480            
481            if (reply != null) {
482                    GraniteContext context = GraniteContext.getCurrentInstance();
483                    if (context instanceof HttpGraniteContext) {
484                        HttpSession session = ((HttpGraniteContext)context).getRequest().getSession(false);
485                        if (session != null)
486                            reply.setHeader("org.granite.sessionId", session.getId());
487                    }
488            }
489            
490            return reply;
491        }

517        private Message handlePingMessage(CommandMessage message) {
518            
519            Channel channel = createChannel();
520            
521            AsyncMessage reply = new AcknowledgeMessage(message);
522            reply.setClientId(channel.getId());
523            Map<String, Object> advice = new HashMap<String, Object>();
524            advice.put(RECONNECT_INTERVAL_MS_KEY, Long.valueOf(gravityConfig.getReconnectIntervalMillis()));
525            advice.put(RECONNECT_MAX_ATTEMPTS_KEY, Long.valueOf(gravityConfig.getReconnectMaxAttempts()));
526            reply.setBody(advice);
527            reply.setDestination(message.getDestination());
528    
529            log.debug("handshake.handle: reply=%s", reply);
530    
531            return reply;
532        }

534        private Message handleSecurityMessage(CommandMessage message) {
535            GraniteConfig config = GraniteContext.getCurrentInstance().getGraniteConfig();
536    
537            Message response = null;
538    
539            if (!config.hasSecurityService())
540                log.warn("Ignored security operation (no security settings in granite-config.xml): %s", message);
541            else {
542                SecurityService securityService = config.getSecurityService();
543                try {
544                    if (message.isLoginOperation())
545                        securityService.login(message.getBody());
546                    else
547                        securityService.logout();
548                }
549                catch (Exception e) {
550                    if (e instanceof SecurityServiceException)
551                        log.debug(e, "Could not process security operation: %s", message);
552                    else
553                        log.error(e, "Could not process security operation: %s", message);
554                    response = new ErrorMessage(message, e, true);
555                }
556            }
557    
558            if (response == null) {
559                response = new AcknowledgeMessage(message, true);
560                // For SDK 2.0.1_Hotfix2.
561                if (message.isSecurityOperation())
562                    response.setBody("success");
563            }
564    
565            return response;
566        }

577        private Message handleDisconnectMessage(CommandMessage message) {
578            Channel client = getChannel((String)message.getClientId());
579            if (client == null)
580                return handleUnknownClientMessage(message);
581    
582            removeChannel(client.getId());
583    
584            AcknowledgeMessage reply = new AcknowledgeMessage(message);
585            reply.setDestination(message.getDestination());
586            reply.setClientId(client.getId());
587            return reply;
588        }

590        private Message handleSubscribeMessage(CommandMessage message) {
591    
592            GraniteContext context = GraniteContext.getCurrentInstance();
593    
594            // Get and check destination.
595            Destination destination = context.getServicesConfig().findDestinationById(
596                message.getMessageRefType(),
597                message.getDestination()
598            );
599    
600            if (destination == null)
601                return getInvalidDestinationError(message);
602    
603    
604            GravityInvocationContext invocationContext = new GravityInvocationContext(message, destination);
605    
606            // Check security 1 (destination).
607            if (destination.getSecurizer() instanceof GravityDestinationSecurizer) {
608                try {
609                    ((GravityDestinationSecurizer)destination.getSecurizer()).canSubscribe(invocationContext);
610                } catch (Exception e) {
611                    return new ErrorMessage(message, e);
612                }
613            }
614    
615            // Check security 2 (security service).
616            GraniteConfig config = context.getGraniteConfig();
617            if (config.hasSecurityService()) {
618                try {
619                    config.getSecurityService().authorize(invocationContext);
620                } catch (Exception e) {
621                    return new ErrorMessage(message, e);
622                }
623            }
624    
625            // Subscribe...
626            Channel channel = getChannel((String)message.getClientId());
627            if (channel == null)
628                return handleUnknownClientMessage(message);
629    
630            String subscriptionId = (String)message.getHeader(AsyncMessage.DESTINATION_CLIENT_ID_HEADER);
631            if (subscriptionId == null) {
632                subscriptionId = UUIDUtil.randomUUID();
633                message.setHeader(AsyncMessage.DESTINATION_CLIENT_ID_HEADER, subscriptionId);
634            }
635            
636            HttpSession session = null;
637            if (context instanceof HttpGraniteContext)
638                    session = ((HttpGraniteContext)context).getSession(false);
639            
640            if (session != null && Boolean.TRUE.toString().equals(destination.getProperties().get("session-selector"))) {
641                    String selector = (String)session.getAttribute("org.granite.gravity.selector." + destination.getId());
642                    log.debug("Session selector found in session %s: %s", session.getId(), selector);
643                    if (selector != null)
644                            message.setHeader(CommandMessage.SELECTOR_HEADER, selector);
645            }
646    
647            ServiceAdapter adapter = adapterFactory.getServiceAdapter(message);
648            
649            AsyncMessage reply = (AsyncMessage)adapter.manage(channel, message);
650            
651            postManage(channel);
652    
653            reply.setDestination(message.getDestination());
654            reply.setClientId(channel.getId());
655            reply.getHeaders().putAll(message.getHeaders());
656    
657            if (session != null && message.getDestination() != null) {
658                    session.setAttribute("org.granite.gravity.channel.clientId." + message.getDestination(), channel.getId());
659                    session.setAttribute("org.granite.gravity.channel.subscriptionId." + message.getDestination(), subscriptionId);
660            }
661    
662            return reply;
663        }

665        private Message handleUnsubscribeMessage(CommandMessage message) {
666            Channel channel = getChannel((String)message.getClientId());
667            if (channel == null)
668                return handleUnknownClientMessage(message);
669    
670            AsyncMessage reply = null;
671    
672            ServiceAdapter adapter = adapterFactory.getServiceAdapter(message);
673            
674            reply = (AcknowledgeMessage)adapter.manage(channel, message);
675            
676            postManage(channel);
677    
678            reply.setDestination(message.getDestination());
679            reply.setClientId(channel.getId());
680            reply.getHeaders().putAll(message.getHeaders());
681    
682            return reply;
683        }

692        private Message handlePublishMessage(AsyncMessage message, Channel channel) {
693    
694            GraniteContext context = GraniteContext.getCurrentInstance();
695    
696            // Get and check destination.
697            Destination destination = context.getServicesConfig().findDestinationById(
698                message.getClass().getName(),
699                message.getDestination()
700            );
701    
702            if (destination == null)
703                return getInvalidDestinationError(message);
704    
705    
706            GravityInvocationContext invocationContext = new GravityInvocationContext(message, destination);
707    
708            // Check security 1 (destination).
709            if (destination.getSecurizer() instanceof GravityDestinationSecurizer) {
710                try {
711                    ((GravityDestinationSecurizer)destination.getSecurizer()).canPublish(invocationContext);
712                } catch (Exception e) {
713                    return new ErrorMessage(message, e, true);
714                }
715            }
716    
717            // Check security 2 (security service).
718            GraniteConfig config = context.getGraniteConfig();
719            if (config.hasSecurityService() && context instanceof HttpGraniteContext) {
720                try {
721                    config.getSecurityService().authorize(invocationContext);
722                } catch (Exception e) {
723                    return new ErrorMessage(message, e, true);
724                }
725            }
726    
727            // Publish...
728            Channel fromChannel = channel;
729            if (fromChannel == null)
730                    fromChannel = getChannel((String)message.getClientId());
731            if (fromChannel == null)
732                return handleUnknownClientMessage(message);
733    
734            ServiceAdapter adapter = adapterFactory.getServiceAdapter(message);
735            
736            AsyncMessage reply = (AsyncMessage)adapter.invoke(fromChannel, message);
737    
738            reply.setDestination(message.getDestination());
739            reply.setClientId(fromChannel.getId());
740    
741            return reply;
742        }

744        private Message handleUnknownClientMessage(Message message) {
745            ErrorMessage reply = new ErrorMessage(message, true);
746            reply.setFaultCode("Server.Call.UnknownClient");
747            reply.setFaultString("Unknown client");
748            return reply;
749        }

754        private ErrorMessage getInvalidDestinationError(Message message) {
755    
756            String messageType = message.getClass().getName();
757            if (message instanceof CommandMessage)
758                messageType += '[' + ((CommandMessage)message).getMessageRefType() + ']';
759    
760            ErrorMessage reply = new ErrorMessage(message, true);
761            reply.setFaultCode("Server.Messaging.InvalidDestination");
762            reply.setFaultString(
763                "No configured destination for id: " + message.getDestination() +
764                " and message type: " + messageType
765            );
766            return reply;
767        }
