062    public class EclipseLinkExternalizer extends DefaultExternalizer {

064            private static final Logger log = Logger.getLogger(EclipseLinkExternalizer.class);
065    
066        @Override
067        public Object newInstance(String type, ObjectInput in)
068            throws IOException, ClassNotFoundException, InstantiationException, InvocationTargetException, IllegalAccessException {
069    
070            // If type is not an entity (@Embeddable for example), we don't read initialized/detachedState
071            // and we fall back to DefaultExternalizer behavior.
072            Class<?> clazz = ClassUtil.forName(type);
073            if (!isRegularEntity(clazz))
074                return super.newInstance(type, in);
075    
076            // Read initialized flag.
077            boolean initialized = ((Boolean)in.readObject()).booleanValue();
078            
079            // Read detached state.
080            @SuppressWarnings("unused")
081                    String detachedState = (String)in.readObject();
082            
083            // New or initialized entity.
084            if (initialized)
085                    return super.newInstance(type, in);
086    
087            // Pseudo-proxy (uninitialized entity).
088            Object id = in.readObject();
089            if (id != null && (!clazz.isAnnotationPresent(IdClass.class) || !clazz.getAnnotation(IdClass.class).value().equals(id.getClass())))
090                    throw new RuntimeException("Id for EclipseLink pseudo-proxy should be null or IdClass (" + type + ")");
091            
092            return new EclipseLinkValueHolder();
093        }
094    
095        @Override
096        public void readExternal(Object o, ObjectInput in) throws IOException, ClassNotFoundException, IllegalAccessException {
097            // Ignore EclipseLink proxies
098            if (o instanceof EclipseLinkValueHolder)
099                    return;
100    
101            // @Embeddable or others...
102            if (!isRegularEntity(o.getClass())) {
103                    log.debug("Delegating non regular entity reading to DefaultExternalizer...");
104                super.readExternal(o, in);
105            }
106            // Regural @Entity or @MappedSuperclass
107            else {
108                GraniteConfig config = GraniteContext.getCurrentInstance().getGraniteConfig();
109    
110                Converters converters = config.getConverters();
111                ClassGetter classGetter = config.getClassGetter();
112                Class<?> oClass = classGetter.getClass(o);
113    
114                List<Property> fields = findOrderedFields(oClass);
115                log.debug("Reading entity %s with fields %s", oClass.getName(), fields);
116                Map<String, Property> topLinkFields = new HashMap<String, Property>();
117                for (Property field : fields) {
118                    if (field.getType() instanceof Class<?> && ValueHolderInterface.class.isAssignableFrom((Class<?>)field.getType())) {
119                        topLinkFields.put(field.getName(), field);
120                    }
121                    else {
122                        Object value = in.readObject();
123    
124                        if (value instanceof ValueHolderInterface) {
125                            topLinkFields.get("_persistence_" + field.getName() + "_vh").setProperty(o, value, false);
126                        }
127                        else if (!(field instanceof MethodProperty && field.isAnnotationPresent(ExternalizedProperty.class))) { 
128                            if (value instanceof AbstractExternalizablePersistentCollection)
129                                value = newIndirectCollection((AbstractExternalizablePersistentCollection)value, field.getType());
130                            else
131                                value = converters.convert(value, field.getType());
132                            
133                            field.setProperty(o, value, false);
134                        }
135                    }
136                }
137            }
138        }

141        protected IndirectContainer newIndirectCollection(AbstractExternalizablePersistentCollection value, Type target) {
142            final boolean initialized = value.isInitialized();
143                    final Object[] content = value.getContent();
144            
145                    IndirectContainer coll = null;
146                    if (value instanceof ExternalizablePersistentSet) {
147                if (initialized) {
148                    if (content != null) {
149                                    Set<?> set = ((ExternalizablePersistentSet)value).getContentAsSet(target);
150                        coll = new IndirectSet(set);
151                    }
152                } 
153                else
154                    coll = new IndirectSet();
155            }
156                    else if (value instanceof ExternalizablePersistentList) {
157                    if (initialized) {
158                        if (content != null) {
159                            List<?> list = ((ExternalizablePersistentList)value).getContentAsList(target);
160                            coll = new IndirectList(list);
161                        }
162                    }
163                    else
164                        coll = new IndirectList();
165                    }
166                    else if (value instanceof ExternalizablePersistentMap) {
167                    if (initialized) {
168                        if (content != null) {
169                            Map<?, ?> map = ((ExternalizablePersistentMap)value).getContentAsMap(target);
170                            coll = new IndirectMap(map);
171                        }
172                    }
173                    else
174                        coll = new IndirectMap();
175                    }
176                    else {
177                            throw new RuntimeException("Illegal externalizable persitent class: " + value);
178                    }
179            
180            return coll;
181        }

184        @Override
185        public void writeExternal(Object o, ObjectOutput out) throws IOException, IllegalAccessException {
186    
187            ClassGetter classGetter = GraniteContext.getCurrentInstance().getGraniteConfig().getClassGetter();
188            Class<?> oClass = classGetter.getClass(o);
189    
190            if (o instanceof EclipseLinkProxy) {            
191                EclipseLinkProxy proxy = (EclipseLinkProxy)o;
192    
193                // Only write initialized flag, null detachedState & null id if proxy is uninitialized.
194                    String description = proxy.getProxiedClass().getName();
195                    log.debug("Writing uninitialized EclipseLink ValueHolder %s", description);
196                    
197                    // Write initialized flag.
198                    out.writeObject(Boolean.FALSE);
199                    // Write detachedState.
200                out.writeObject(null);
201                    // Write id.
202                out.writeObject(null);
203    
204                return;
205            }
206    
207            if (!isRegularEntity(o.getClass())) { // @Embeddable or others...
208                    log.debug("Delegating non regular entity writing to DefaultExternalizer...");
209                super.writeExternal(o, out);
210            }
211            else {
212                    // Write initialized flag.
213                    out.writeObject(Boolean.TRUE);
214                    // Write detachedState.
215                out.writeObject(null);
216    
217                // Externalize entity fields.
218                List<Property> fields = findOrderedFields(oClass);
219                List<String> lazyFieldNames = new ArrayList<String>();
220                
221                log.debug("Writing entity %s with fields %s", o.getClass().getName(), fields);
222                for (Property field : fields) {
223                    if (!(field.getType() instanceof Class<?> && ValueHolderInterface.class.isAssignableFrom((Class<?>)field.getType()))) {
224                        if (lazyFieldNames.contains(field.getName())) {
225                            EclipseLinkProxy proxy = new EclipseLinkProxy((Class<?>)field.getType());
226                            out.writeObject(proxy);
227                        }
228                        else {
229                            Object value = field.getProperty(o);
230            
231                            if (value instanceof IndirectContainer)
232                                    value = newExternalizableCollection((IndirectContainer)value);
233                            
234                            out.writeObject(value);
235                        }
236                    }
237                    else {
238                        ValueHolderInterface vh = (ValueHolderInterface)field.getProperty(o);
239                        if (!vh.isInstantiated())
240                            lazyFieldNames.add(field.getName().substring("_persistence_".length(), field.getName().length()-3));
241                    }
242                }
243            }
244        }
245        
246        protected AbstractExternalizablePersistentCollection newExternalizableCollection(IndirectContainer value) {
247            AbstractExternalizablePersistentCollection coll = null;
248            boolean initialized = value.isInstantiated();
249            
250            if (value instanceof IndirectSet) {
251                coll = new ExternalizablePersistentSet(initialized ? ((IndirectSet)value).toArray() : null, initialized, false);
252            }
253            else if (value instanceof IndirectList) {
254                coll = new ExternalizablePersistentList(initialized ? ((IndirectList)value).toArray() : null, initialized, false);
255            }
256            else if (value instanceof IndirectMap) {
257                Object[] content = null;
258                
259                if (initialized) {
260                    content = new Object[((IndirectMap)value).size()];
261                    int index = 0;
262                    @SuppressWarnings("unchecked")
263                    Set<Map.Entry<?, ?>> entries = ((IndirectMap)value).entrySet();
264                    for (Map.Entry<?, ?> entry : entries)
265                        content[index++] = new Object[]{entry.getKey(), entry.getValue()};
266                }
267                
268                coll = new ExternalizablePersistentMap(content, initialized, false);
269            }
270            else {
271                throw new UnsupportedOperationException("Unsupported EclipseLink collection type: " + value);
272            }
273            
274            return coll;
275        }

278        @Override
279        public int accept(Class<?> clazz) {
280            return (
281                clazz.isAnnotationPresent(Entity.class) ||
282                clazz.isAnnotationPresent(MappedSuperclass.class) ||
283                clazz.isAnnotationPresent(Embeddable.class)
284            ) ? 1 : -1;
285        }
286    
287        protected boolean isRegularEntity(Class<?> clazz) {
288            return clazz.isAnnotationPresent(Entity.class) || clazz.isAnnotationPresent(MappedSuperclass.class);
289        }
