021    package org.granite.util;
022    
023    import java.io.IOException;
024    import java.io.InputStream;
025    import java.io.OutputStream;
026    
027    /**
028     * @author Franck WOLFF
029     */
030    public class StreamGobbler extends Thread {
031    
032        private final InputStream is;
033        private final OutputStream os;
034    
035        public StreamGobbler(InputStream is, OutputStream os) {
036            this.is = is;
037            this.os = os;
038        }
039    
040        @Override
041        public void run() {
042            try {
043                for (int b = is.read(); b != -1; b = is.read())
044                    os.write(b & 0xFF);
045            } catch (IOException e)  {
046                throw new RuntimeException(e);
047            } finally {
048                try {
049                    os.flush();
050                    os.close();
051                } catch (Exception e) {
052                    throw new RuntimeException(e);
053                }
054            }
055        }
056    }
